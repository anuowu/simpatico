65  * Generate, attempt and accept or reject a move.
66  */
67  bool AtomDisplaceMove::move()
68  {
69  Vector oldPos;
70  double newEnergy, oldEnergy;
71  Molecule* molPtr;
72  Atom* atomPtr;
73  int iAtom;
74 
75  incrementNAttempt();
76 
77  // Choose a molecule and atom at random
78  molPtr = &(system().randomMolecule(speciesId_));
79  iAtom = random().uniformInt(0, molPtr->nAtom());
80  atomPtr = &molPtr->atom(iAtom);
81 
82  // Calculate current pair energy and store old position
83  oldPos = atomPtr->position();
84  oldEnergy = system().atomPotentialEnergy(*atomPtr);
85 
86  for (int j = 0; j < Dimension; ++j) {
87  atomPtr->position()[j] += random().uniform(-delta_, delta_);
88  }
89  boundary().shift(atomPtr->position());
90  newEnergy = system().atomPotentialEnergy(*atomPtr);
91 
92  // Decide whether to accept forward move
93  bool accept = random().metropolis(boltzmann(newEnergy - oldEnergy));
94 
95  if (accept) {
96  #ifndef SIMP_NOPAIR
97  system().pairPotential().updateAtomCell(*atomPtr);
98  #endif
99  incrementNAccept();
100  } else {
101  atomPtr->position() = oldPos;
102  }
103 
104  return accept;
105  }
