89  */
90  template < class Interaction >
91  double McExternalPerturbation<Interaction>::derivative(int i) const
92  {
93  // Preconditions
94  if (i > nParameters_) {
95  UTIL_THROW("perturbation parameter index is out of bounds");
96  }
97 
98  if (fabs(parameter_[i] - parameter(i)) > 1.0E-8) {
99  UTIL_THROW("Perturbation parameter is not set correctly");
100  }
101 
102  double energy;
103  System::MoleculeIterator molIter;
104  Molecule::AtomIterator atomIter;
105 
106  energy = 0.0;
107 
108  for (int iSpec=0; iSpec < system().simulation().nSpecies(); ++iSpec){
109  for (system().begin(iSpec, molIter); molIter.notEnd(); ++molIter){
110  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
111  energy += interaction().energy(atomIter->position(), atomIter->typeId());
112 
113  }
114  }
115  }
116 
117  // Multiply by the temperature factor.
118  if (system().energyEnsemble().isIsothermal()) {
119  energy *= system().energyEnsemble().beta();
120  } else {
121  UTIL_THROW("Non isothermal ensemble.");
122  }
123 
124  return energy/parameter_[i];
125  }
