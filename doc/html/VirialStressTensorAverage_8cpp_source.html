138  /*
139  * Sample the stress tensor.
140  */
141  void VirialStressTensorAverage::sample(long iStep)
142  {
143  if (isAtInterval(iStep)) {
144  Simulation& sys = simulation();
145  sys.computeVirialStress();
146  sys.computeKineticStress();
147  if (sys.domain().isMaster()) {
148  Tensor virial = sys.virialStress();
149  Tensor kinetic = sys.kineticStress();
150  Tensor total = total.add(virial, kinetic);
151  sxxAccumulator_.sample(virial(0,0));
152  sxyAccumulator_.sample(virial(0,1));
153  sxzAccumulator_.sample(virial(0,2));
154  syxAccumulator_.sample(virial(1,0));
155  syyAccumulator_.sample(virial(1,1));
156  syzAccumulator_.sample(virial(1,2));
157  szxAccumulator_.sample(virial(2,0));
158  szyAccumulator_.sample(virial(2,1));
159  szzAccumulator_.sample(virial(2,2));
160  }
161  ++nSample_;
162  }
