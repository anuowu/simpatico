69  * Pop and process the next molecule on the workStack (private).
70  */
71  void ClusterIdentifier::processNextMolecule(Cluster& cluster)
72  {
73  CellList::NeighborArray neighborArray;
74  Molecule::AtomIterator atomIter;
75  ClusterLink* thisLinkPtr;
76  ClusterLink* otherLinkPtr;
77  Atom* otherAtomPtr;
78  Boundary& boundary = system().boundary();
79  double cutoffSq = cutoff_*cutoff_;
80  double rsq;
81  int thisMolId, otherMolId, otherClusterId;
82 
83  // Pop this molecule off the stack
84  thisLinkPtr = &workStack_.pop();
85  thisMolId = thisLinkPtr->molecule().id();
86  if (thisLinkPtr->clusterId() != cluster.id()) {
87  UTIL_THROW("Top ClusterLink not marked with this cluster id");
88  }
89 
90  /*
91  * Loop over atoms of this molecule.
92  * For each atom of type atomTypeId_, find neighboring molecules.
93  * Add each new neighbor to the cluster, and to the workStack.
94  */
95  thisLinkPtr->molecule().begin(atomIter);
96  for ( ; atomIter.notEnd(); ++atomIter) {
97  if (atomIter->typeId() == atomTypeId_) {
98  cellList_.getNeighbors(atomIter->position(), neighborArray);
99  for (int i = 0; i < neighborArray.size(); i++) {
100  otherAtomPtr = neighborArray[i];
101  otherMolId = otherAtomPtr->molecule().id();
102  if (otherMolId != thisMolId) {
103  rsq = boundary.distanceSq(atomIter->position(),
104  otherAtomPtr->position());
105  if (rsq < cutoffSq) {
106  otherLinkPtr = &(links_[otherMolId]);
107  assert(&otherLinkPtr->molecule() ==
108  &otherAtomPtr->molecule());
109  otherClusterId = otherLinkPtr->clusterId();
110  if (otherClusterId == -1) {
111  cluster.addLink(*otherLinkPtr);
112  workStack_.push(*otherLinkPtr);
113  } else
114  if (otherClusterId != cluster.id()) {
115  UTIL_THROW("Cluster Clash!");
116  }
117  }
118  }
119  } // neighbor atoms
120  }
121  } // atoms
122

126  * Identify all clusters in the system.
127  */
128  void ClusterIdentifier::identifyClusters()
129  {
130 
131  // Initialize all data structures:
132  // Setup a grid of empty cells
133  cellList_.setup(system().boundary(), cutoff_);
134  // Clear clusters array and all links
135  clusters_.clear();
136  for (int i = 0; i < links_.capacity(); ++i) {
137  links_[i].clear();
138  }
139 
140  // Build the cellList, associate Molecule with ClusterLink.
141  // Iterate over molecules of species speciesId_
142  System::MoleculeIterator molIter;
143  Molecule::AtomIterator atomIter;
144  system().begin(speciesId_, molIter);
145  for ( ; molIter.notEnd(); ++molIter) {
146 
147  // Associate this Molecule with a ClusterLink
148  links_[molIter->id()].setMolecule(*molIter.get());
149 
150  // Add atoms of type = atomTypeId_ to the CellList
151  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
152  if (atomIter->typeId() == atomTypeId_) {
153  system().boundary().shift(atomIter->position());
154  cellList_.addAtom(*atomIter);
155  }
156 
157  }
158  }
159 
160  // Identify all clusters
161  Cluster* clusterPtr;
162  ClusterLink* linkPtr;
163  int clusterId = 0;
164  system().begin(speciesId_, molIter);
165  for ( ; molIter.notEnd(); ++molIter) {
166 
167  // Find the link with same index as this molecule
168  linkPtr = &(links_[molIter->id()]);
169  assert (&(linkPtr->molecule()) == molIter.get());
170 
171  // If this link is not in a cluster, begin a new cluster
172  if (linkPtr->clusterId() == -1) {
173 
174  // Add a new empty cluster to clusters_ array
175  clusters_.resize(clusterId+1);
176  clusterPtr = &clusters_[clusterId];
177  clusterPtr->clear();
178  clusterPtr->setId(clusterId);
179 
180  // Identify molecules in this cluster
181  clusterPtr->addLink(*linkPtr);
182  workStack_.push(*linkPtr);
183  while (workStack_.size() > 0) {
184  processNextMolecule(*clusterPtr);
185  }
186 
187  clusterId++;
188  }
189 
190  }

193  isValid();
