40  int nSpecies = simulation().nSpecies();
41  int speciesCapacity = 0;
42  int iSpec,iMol;
43  Species* speciesPtr;
44  Molecule* molPtr;
45 
46  // Add molecules to system
47  nAtomTotal_ = 0;
48  for (iSpec = 0; iSpec < nSpecies; ++iSpec) {
49  speciesPtr = &simulation().species(iSpec);
50  speciesCapacity = speciesPtr->capacity();
51 
52  for (iMol = 0; iMol < speciesCapacity; ++iMol) {
53  molPtr = &(simulation().getMolecule(iSpec));
54  system().addMolecule(*molPtr);
55  }
56  nAtomTotal_ += speciesCapacity*speciesPtr->nAtom();
57  }
