110  /*
111  * Check consistency of data.
112  */
113  bool OrthorhombicBoundary::isValid()
114  {
115  double dot;
116  double twoPi = 2.0*Constants::Pi;
117  int i, j;
118 
119  OrthoRegion::isValid();
120  for (i = 0; i < Dimension; ++i) {
121  if (!feq(minima_[i], 0.0)) {
122  UTIL_THROW("minima_[i] != 0");
123  }
124  if (!feq(lengths_[i]*invLengths_[i], 1.0)) {
125  UTIL_THROW("invLengths_[i]*lengths_[i] != 1.0");
126  }
127  if (!feq(lengths_[i], bravaisBasisVectors_[i][i])) {
128  UTIL_THROW("bravaisBasisVectors_[i][i] != lengths_[i]");
129  }
130  for (j = 0; j < Dimension; ++j) {
131  dot = bravaisBasisVectors_[i].dot(reciprocalBasisVectors_[j]);
132  if (i == j) {
133  if (!feq(dot, twoPi)) {
134  UTIL_THROW("a[i].b[i] != twoPi");
135  }
136  } else {
137  if (!feq(dot, 0.0)) {
138  UTIL_THROW("a[i].b[j] != 0 for i != j");
139  }
140  if (!feq(bravaisBasisVectors_[i][j], 0.0)) {
141  UTIL_THROW("Nonzero off-diagonal element of bravaisBasisVectors_");
142  }
143  if (!feq(reciprocalBasisVectors_[i][j], 0.0)) {
144  UTIL_THROW("Nonzero off-diagonal element of reciprocalBasisVectors_");
145  }
146  }
147  }
148  }
149  return true;
150  }
