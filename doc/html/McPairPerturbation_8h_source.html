191  /*
192  * Return pair energy for unlike pairs / (kT*epsilon(0,1))
193  */
194  template < class Interaction >
195  double McPairPerturbation<Interaction>::derivative(int i) const
196  {
197  // Preconditions
198  if (i > nParameters_) {
199  UTIL_THROW("perturbation parameter index is out of bounds");
200  }
201  if (fabs(parameter_[i] - parameter(i)) > 1.0E-8) {
202  UTIL_THROW("Perturbation parameter is not set correctly");
203  }
204  if (!system().energyEnsemble().isIsothermal()) {
205  UTIL_THROW("Non isothermal ensemble for McPairPerturbation.");
206  }
207 
208  double energy, rsq;
209  Atom *jAtomPtr, *kAtomPtr;
210  int nNeighbor, nInCell;
211  int ic, nc, j, k, jId, kId, jType, kType;
212 
213  // Loop over cells
214  energy = 0.0;
215  nc = system().pairPotential().cellList().totCells();
216  for (ic = 0; ic < nc; ++ic) {
217 
218  // Get array of neighbors_
219  system().pairPotential().cellList().getCellNeighbors(ic, neighbors_, nInCell);
220  nNeighbor = neighbors_.size();
221 
222  // Loop over primary atoms in this cell
223  for (j = 0; j < nInCell; ++j) {
224  jAtomPtr = neighbors_[j];
225  jId = jAtomPtr->id();
226  jType = jAtomPtr->typeId();
227 
228  // Loop over secondary atoms in this and neighboring cells
229  for (k = 0; k < nNeighbor; ++k) {
230  kAtomPtr = neighbors_[k];
231  kId = kAtomPtr->id();
232  kType = kAtomPtr->typeId();
233 
234  // Count each pair only once
235  if (kId > jId && jType != kType) {
236 
237  // Exclude masked pairs
238  if (!jAtomPtr->mask().isMasked(*kAtomPtr)) {
239 
240  rsq = system().boundary().
241  distanceSq(jAtomPtr->position(), kAtomPtr->position());
242 
243  energy += interaction().
244  energy(rsq, jAtomPtr->typeId(), kAtomPtr->typeId());
245 
246  }
247 
248  }
249 
250  } // secondary atoms
251 
252  } // primary atoms
253 
254  } // cells
255 
256  // Multiply by the temperature factor.
257  if (system().energyEnsemble().isIsothermal()) {
258  energy *= system().energyEnsemble().beta();
259  } else {
260  UTIL_THROW("Non isothermal ensemble for McPairPerturbation.");
261  }
262 
263  return energy/parameter_[i];
264  }

266 }
