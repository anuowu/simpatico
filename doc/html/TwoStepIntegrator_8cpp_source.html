41  /*
42  * Run integrator for nStep steps.
43  */
44  void TwoStepIntegrator::run(int nStep)
45  {
46  // Precondition
47  if (atomStorage().isCartesian()) {
48  UTIL_THROW("Error: Atom coordinates are Cartesian");
49  }
50 
51  // References to managers
52  #ifdef DDMD_MODIFIERS
53  ModifierManager& modifierManager = simulation().modifierManager();
54  #endif
55  AnalyzerManager& analyzerManager = simulation().analyzerManager();
56 
57  // Unset all stored computations.
58  simulation().modifySignal().notify();
59 
60  // Recompute nAtomTotal.
61  atomStorage().unsetNAtomTotal();
62  atomStorage().computeNAtomTotal(domain().communicator());
63 
64  // Setup required before main loop (atoms, ghosts, groups, forces, etc)
65  // Atomic coordinates are Cartesian on exit from Integrator::setup().
66  // Atomic forces should be set on exit from Integrator::setup().
67  setup();
68  #ifdef DDMD_MODIFIERS
69  modifierManager.setup();
70  #endif
71  analyzerManager.setup();
72 
73  // Main MD loop
74  timer().start();
75  exchanger().timer().start();
76  int beginStep = iStep_;
77  int endStep = iStep_ + nStep;
78  bool needExchange;
79  for ( ; iStep_ < endStep; ++iStep_) {
80 
81  // Atomic coordinates must be Cartesian on entry to loop body.
82  if (!atomStorage().isCartesian()) {
83  UTIL_THROW("Error: Atomic coordinates are not Cartesian");
84  }
85 
86  // Sample analyzers, if scheduled.
87  analyzerManager.sample(iStep_);
88 
89  // Write restart file, if scheduled.
90  if (saveInterval() > 0) {
91  if (iStep_ % saveInterval() == 0) {
92  if (iStep_ > beginStep) {
93  simulation().save(saveFileName());
94  }
95  }
96  }
97  timer().stamp(ANALYZER);
98 
99  #ifdef DDMD_MODIFIERS
100  modifierManager.preIntegrate1(iStep_);
101  timer().stamp(MODIFIER);
102  #endif
103 
104  // First step of integration: Update positions, half velocity
105  integrateStep1();
106  timer().stamp(INTEGRATE1);
107 
108  // Unset precomputed values of all energies, stresses, etc.
109  simulation().modifySignal().notify();
110 
111  #ifdef DDMD_INTEGRATOR_DEBUG
112  // Sanity check
113  simulation().isValid();
114  timer().stamp(DEBUG);
115  #endif
116 
117  #ifdef DDMD_MODIFIERS
118  modifierManager.postIntegrate1(iStep_);
119  timer().stamp(MODIFIER);
120  #endif
121 
122  // Check if exchange and reneighboring is necessary
123  // Note: Integrate::isExchangeNeeded uses timer.
124  needExchange = isExchangeNeeded(pairPotential().skin());
125 
126  if (!atomStorage().isCartesian()) {
127  UTIL_THROW("Error: atomic coordinates are not Cartesian");
128  }
129 
130  // Exchange atoms if necessary
131  if (needExchange) {
132 
133  #ifdef DDMD_MODIFIERS
134  modifierManager.preTransform(iStep_);
135  timer().stamp(MODIFIER);
136  #endif
137 
138  // Transform to scaled [0,1] coordinates
139  atomStorage().clearSnapshot();
140  atomStorage().transformCartToGen(boundary());
141  timer().stamp(Integrator::TRANSFORM_F);
142 
143  #ifdef DDMD_MODIFIERS
144  modifierManager.preExchange(iStep_);
145  timer().stamp(MODIFIER);
146  #endif
147 
148  // Exchange atom ownership, reidentify ghosts
149  exchanger().exchange();
150  timer().stamp(Integrator::EXCHANGE);
151 
152  #ifdef DDMD_MODIFIERS
153  modifierManager.postExchange(iStep_);
154  timer().stamp(MODIFIER);
155  #endif
156 
157  // Build cell list
158  pairPotential().buildCellList();
159  timer().stamp(Integrator::CELLLIST);
160 
161  // Transform from scaled [0,1] to Cartesian coordinates.
162  atomStorage().transformGenToCart(boundary());
163  timer().stamp(Integrator::TRANSFORM_R);
164 
165  // Build pair list
166  atomStorage().makeSnapshot();
167  pairPotential().buildPairList();
168  timer().stamp(Integrator::PAIRLIST);
169 
170  #ifdef DDMD_MODIFIERS
171  modifierManager.postNeighbor(iStep_);
172  timer().stamp(MODIFIER);
173  #endif
174 
175  } else { // Update step (no exchange)
176 
177  #ifdef DDMD_MODIFIERS
178  modifierManager.preUpdate(iStep_);
179  timer().stamp(MODIFIER);
180  #endif
181 
182  // Update all ghost atom positions
183  exchanger().update();
184  timer().stamp(UPDATE);
185 
186  #ifdef DDMD_MODIFIERS
187  modifierManager.postUpdate(iStep_);
188  timer().stamp(MODIFIER);
189  #endif
190 
191  }
192  simulation().exchangeSignal().notify();
193 
194  #ifdef DDMD_INTEGRATOR_DEBUG
195  // Sanity check
196  simulation().isValid();
197  timer().stamp(DEBUG);
198  #endif
199 
200  if (!atomStorage().isCartesian()) {
201  UTIL_THROW("Error: Atomic coordinates are not Cartesian");
202  }
203 
204  #ifdef DDMD_MODIFIERS
205  modifierManager.preForce(iStep_);
206  timer().stamp(MODIFIER);
207  #endif
208 
209  // Calculate forces:
210  // Calculate new forces for all local atoms. If constant pressure
211  // ensemble (not rigid), also calculate the virial stress. Both
212  // methods use the timer() internall, and both send the modifyForce
213  // signal.
214  if (simulation().boundaryEnsemble().isRigid()) {
215  computeForces();
216  } else {
217  computeForcesAndVirial();
218  }
219 
220  #ifdef DDMD_MODIFIERS
221  modifierManager.postForce(iStep_);
222  timer().stamp(MODIFIER);
223  #endif
224 
225  // 2nd step of velocity-Verlet integration. This finishes the velocity
226  // update, and normally calls simulation().velocitySignal().notify()
227  integrateStep2();
228  timer().stamp(INTEGRATE2);
229 
230  #ifdef DDMD_INTEGRATOR_DEBUG
231  // Sanity check
232  simulation().isValid();
233  timer().stamp(DEBUG);
234  #endif
235 
236  #ifdef DDMD_MODIFIERS
237  modifierManager.endOfStep(iStep_);
238  timer().stamp(MODIFIER);
239  #endif
240 
241  }
242  exchanger().timer().stop();
243  timer().stop();
244 
245  // Final analyzers and restart file, if scheduled.
246  analyzerManager.sample(iStep_);
247  if (saveInterval() > 0) {
248  if (iStep_ % saveInterval() == 0) {
249  simulation().save(saveFileName());
250  }
251  }
252 
253  // Transform to scaled coordinates, in preparation for the next run.
254  atomStorage().transformCartToGen(boundary());
255 
256  if (domain().isMaster()) {
257  Log::file() << std::endl;
258  }
259 
260  }
261 
262 }
