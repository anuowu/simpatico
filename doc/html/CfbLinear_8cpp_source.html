113  /*
114  * Configuration bias algorithm for deleting one atom from chain end.
115  */
116  void
117  CfbLinear::deleteAtom(Molecule& molecule, int atomId,
118  int sign, double &rosenbluth, double &energy)
119  {
120  // sign == -1, shift == 0 -> atomId = 0, 1, 2, ...
121  // sign == +1, shift == 1 -> atomId = nAtom-1, nAtom-2, ...
122  assert(sign == 1 || sign == -1);
123  int shift = (sign == 1) ? 1 : 0;
124  Atom& atom0 = molecule.atom(atomId);
125  Atom& atom1 = molecule.atom(atomId - sign);
126  Vector& pos0 = atom0.position();
127  Vector& pos1 = atom1.position();
128 
129  // Calculate bond length and energy
130  Vector v1; // Vector between atoms 0 and 1
131  Vector u1; // Unit vector parallel to v1
132  double r1, bondEnergy;
133  int bondTypeId = molecule.bond(atomId - shift).typeId();
134  r1 = boundary().distanceSq(pos1, pos0, v1);
135  // Here r1 = bond length squared, v1 = pos1 - pos0
136  bondEnergy = system().bondPotential().energy(r1, bondTypeId);
137  r1 = sqrt(r1); // bond length
138  u1 = v1;
139  u1 /= r1; // unit vector
140 
141  #ifndef INTER_NOPAIR
142  McPairPotential& pairPotential = system().pairPotential();
143  energy = pairPotential.atomEnergy(atom0);
144  #else
145  energy = 0.0;
146  #endif
147 
148  #ifdef INTER_ANGLE
149  Vector v2, u2;
150  double r2, cosTheta;
151  Vector* pos2Ptr;
152  AnglePotential& anglePotential = system().anglePotential();
153  int angleTypeId;
154  bool hasAngle = false;
155  if (hasAngles_) {
156  int atom2Id = atomId - 2*sign;
157  if (shift) {
158  assert(sign == 1);
159  hasAngle = (atom2Id >= 0);
160  } else {
161  assert(sign == -1);
162  hasAngle = (atom2Id < molecule.nAtom());
163  }
164  if (hasAngle) {
165  Atom& atom2 = molecule.atom(atom2Id);
166  pos2Ptr = &(atom2.position());
167  r2 = boundary().distanceSq(*pos2Ptr, pos1, v2); // v2=p2-p1
168  r2 = sqrt(r2); // r2 = |v2| = length of bond 2
169  u2 = v2;
170  u2 /= r2; // unit vector
171  cosTheta = u1.dot(u2);
172  angleTypeId = molecule.bond(atomId - 2*shift).typeId();
173  energy += anglePotential.energy(cosTheta, angleTypeId);
174  }
175  }
176  #endif
177 
178  #ifdef INTER_EXTERNAL
179  ExternalPotential& externalPotential = system().externalPotential();
180  if (hasExternal_) {
181  energy += externalPotential.atomEnergy(atom0);
182  }
183  #endif
184 
185  // Here: energy = total energy - bond energy
186  // Rosenbluth factor = exp(-beta*(pair + angle + external))
187  rosenbluth = boltzmann(energy);
188 
189  // Add bond energy to total energy in current position
190  energy += bondEnergy;
191 
192  // Loop over nTrial - 1 additional trial positions:
193  double trialEnergy;
194  for (int iTrial = 0; iTrial < nTrial_ - 1; ++iTrial) {
195 
196  // Generate trial position
197  random().unitVector(u1);
198  v1 = u1;
199  v1 *= r1;
200  pos0.subtract(pos1, v1);
201  boundary().shift(pos0);
202 
203  // Compute trial energy (excluding bond energy)
204  #ifndef INTER_NOPAIR
205  trialEnergy = system().pairPotential().atomEnergy(atom0);
206  #else
207  trialEnergy = 0.0;
208  #endif
209  #ifdef INTER_ANGLE
210  if (hasAngle) {
211  cosTheta = u1.dot(u2);
212  trialEnergy += anglePotential.energy(cosTheta, angleTypeId);
213  }
214  #endif
215  #ifdef INTER_EXTERNAL
216  if (hasExternal_) {
217  trialEnergy += externalPotential.atomEnergy(atom0);
218  }
219  #endif
220 
221  rosenbluth += boltzmann(trialEnergy);
222  }
223 
224  }

226  /*
227  * Configuration bias algorithm for adding one atom to a chain end.
228  */
229  void
230  CfbLinear::addAtom(Molecule& molecule, Atom& atom0, Atom& atom1, int atomId,
231  int sign, double &rosenbluth, double &energy)
232  {
233  // sign == -1, shift == 0 -> atomId = 0, 1, 2, ...
234  // sign == +1, shift == 1 -> atomId = nAtom-1, nAtom-2, ...
235  assert(sign == 1 || sign == -1);
236  int shift = (sign == 1) ? 1 : 0;
237 
238  assert(atom0.typeId() == molecule.species().atomTypeId(atomId));
239  Vector& pos0 = atom0.position();
240  Vector& pos1 = atom1.position();
241 
242  // Generate a random bond length r1
243  double beta, r1;
244  int bondTypeId, iTrial;
245  bondTypeId = molecule.bond(atomId - shift).typeId();
246  beta = energyEnsemble().beta();
247  r1 = system().bondPotential().randomBondLength(&random(), beta, bondTypeId);
248 
249  #ifndef INTER_NOPAIR
250  McPairPotential& pairPotential = system().pairPotential();
251  #endif
252 
253  #ifdef INTER_ANGLE
254  // Calculate vector v2 = pos2 - pos1, r2 = |v2|
255  Vector v2, u2;
256  double r2, cosTheta;
257  Vector* pos2Ptr;
258  AnglePotential& anglePotential = system().anglePotential();
259  int angleTypeId;
260  bool hasAngle = false;
261  if (hasAngles_) {
262  int atom2Id = atomId - 2*sign;
263  if (shift) {
264  assert(sign == 1);
265  hasAngle = (atom2Id >= 0);
266  } else {
267  assert(sign == -1);
268  hasAngle = (atom2Id < molecule.nAtom());
269  }
270  if (hasAngle) {
271  angleTypeId = molecule.bond(atomId - 2*shift).typeId();
272  Atom* atom2Ptr = &atom1 - sign;
273  pos2Ptr = &(atom2Ptr->position());
274  r2 = boundary().distanceSq(*pos2Ptr, pos1, v2);
275  r2 = sqrt(r2);
276  u2 = v2;
277  u2 /= r2; // unit vector
278  }
279  }
280  #endif
281  #ifdef INTER_EXTERNAL
282  ExternalPotential& externalPotential
283  = system().externalPotential();
284  #endif
285 
286  // Loop over nTrial trial positions:
287  Vector v1, u1;
288  Vector trialPos[MaxTrial_];
289  double trialProb[MaxTrial_], trialEnergy[MaxTrial_];
290  rosenbluth = 0.0;
291  for (iTrial = 0; iTrial < nTrial_; ++iTrial) {
292 
293  // Generate trial bond vector v1 and position pos0
294  random().unitVector(u1);
295  v1 = u1;
296  v1 *= r1;
297  trialPos[iTrial].subtract(pos1, v1);
298  boundary().shift(trialPos[iTrial]);
299  pos0 = trialPos[iTrial];
300 
301  // Compute trial energy (excluding bond energy)
302  #ifndef INTER_NOPAIR
303  trialEnergy[iTrial] = pairPotential.atomEnergy(atom0);
304  #else
305  trialEnergy[iTrial] = 0.0;
306  #endif
307  #ifdef INTER_ANGLE
308  if (hasAngle) {
309  cosTheta = u1.dot(u2);
310  trialEnergy[iTrial] +=
311  anglePotential.energy(cosTheta, angleTypeId);
312  }
313  #endif
314  #ifdef INTER_EXTERNAL
315  if (hasExternal_) {
316  trialEnergy[iTrial] +=
317  externalPotential.atomEnergy(atom0);
318  }
319  #endif
320 
321  // Compute unnormalized probability, increment rosenbluth
322  trialProb[iTrial] = boltzmann(trialEnergy[iTrial]);
323  rosenbluth += trialProb[iTrial];
324  }
325 
326  // Normalize trial probabilities
327  for (iTrial = 0; iTrial < nTrial_; ++iTrial) {
328  trialProb[iTrial] = trialProb[iTrial]/rosenbluth;
329  }
330 
331  // Choose a trial
332  iTrial = random().drawFrom(trialProb, nTrial_);
333 
334  // Calculate total energy for chosen trial, including bond energy
335  energy = trialEnergy[iTrial];
336  energy += system().bondPotential().energy(r1*r1, bondTypeId);
337 
338  // Set position to chosen value
339  pos0 = trialPos[iTrial];
340  }
