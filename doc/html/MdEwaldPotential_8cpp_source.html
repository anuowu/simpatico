116  /*
117  * Generate waves using kCutoff; allocate memories for associated variables.
118  * Generate vectors for image cell in real sapce using rSpaceCutoff.
119  * Comments:
120  * Only half of waves are stored.
121  * The first index is always non-negative.
122  *
123  */
124  void MdEwaldPotential::makeWaves()
125  {
126  Vector b0, b1, b2; // Recprocal basis vectors.
127  Vector q0, q1, q2, q; // Partial and complete wavevectors.
128  Vector kv; // Wavevector (as real vector)
129  double ksq;
130  IntVector maxK, k; // Max and running wave indices.
131  int mink1, mink2; // Minimum k-indices
132  int j;
133 
134  b0 = boundaryPtr_->reciprocalBasisVector(0);
135  b1 = boundaryPtr_->reciprocalBasisVector(1);
136  b2 = boundaryPtr_->reciprocalBasisVector(2);
137 
138  // Get max wave indices and reserve arrays
139  double pi2 = 2.0*Constants::Pi;
140  for (j=0; j < Dimension; ++j) {
141  maxK[j] =
142  ceil(kSpaceCutoff_*boundaryPtr_->bravaisBasisVector(j).abs()/pi2);
143  UTIL_CHECK(maxK[j] > 0);
144  }
145 
146  if (intWaves_.capacity() == 0) {
147  int capacity;
148  capacity = ((2*maxK[0] + 1)*(2*maxK[1] + 1)*(2*maxK[2] + 1) - 1)/2;
149  UTIL_CHECK(capacity > 0);
150  intWaves_.reserve(capacity);
151  ksq_.reserve(capacity);
152  g_.reserve(capacity);
153  rho_.reserve(capacity);
154  fexp0_.reserve(maxK[0] + 1);
155  fexp1_.reserve(2*maxK[1] + 1);
156  fexp2_.reserve(2*maxK[2] + 1);
157  } else {
158  intWaves_.clear();
159  ksq_.clear();
160  g_.clear();
161  rho_.clear();
162  fexp0_.clear();
163  fexp1_.clear();
164  fexp2_.clear();
165  }
166 
167  // Accumulate waves, and wave-related properties.
168  base0_ = 0;
169  upper0_ = -maxK[0];
170  base1_ = maxK[1];
171  upper1_ = -base1_;
172  base2_ = maxK[2];
173  upper2_ = -base2_;
174  double kSpaceCutoffSq = kSpaceCutoff_*kSpaceCutoff_;
175 
176  q0.multiply(b0, -1);
177  for (k[0] = 0; k[0] <= maxK[0]; ++k[0]) {
178 
179  // Note: First index always non-negative.
180  q0 += b0;
181 
182  mink1 = (k[0] == 0 ? 0 : -maxK[1]);
183  q1.multiply(b1, mink1 - 1);
184  q1 += q0;
185  for (k[1] = mink1; k[1] <= maxK[1]; ++k[1]) {
186  q1 += b1;
187 
188  mink2 = (k[0] == 0 && k[1] == 0 ? 1 : -maxK[2]);
189  q.multiply(b2, mink2 - 1);
190  q += q1;
191 
192  for (k[2] = mink2; k[2] <= maxK[2]; ++k[2]) {
193  q += b2;
194 
195  ksq = double(q.square());
196  if (ksq <= kSpaceCutoffSq) {
197 
198  if (k[0] > upper0_) upper0_ = k[0];
199 
200  if (k[1] < base1_ ) base1_ = k[1];
201  if (k[1] > upper1_) upper1_ = k[1];
202 
203  if (k[2] < base2_ ) base2_ = k[2];
204  if (k[2] > upper2_) upper2_ = k[2];
205 
206  intWaves_.append(k);
207  ksq_.append(ksq);
208  g_.append(ewaldInteraction_.kSpacePotential(ksq));
209  }
210 
211  } // for k[2]
212  } // for k[1]
213  } // for k[0]
214 
215  // Resize work arrays
216  UTIL_CHECK(intWaves_.size() > 0);
217  UTIL_CHECK(upper0_ - base0_ + 1 > 0);
218  UTIL_CHECK(upper1_ - base1_ + 1 > 0);
219  UTIL_CHECK(upper2_ - base2_ + 1 > 0);
220  rho_.resize(intWaves_.size());
221  fexp0_.resize(upper0_ - base0_ + 1);
222  fexp1_.resize(upper1_ - base1_ + 1);
223  fexp2_.resize(upper2_ - base2_ + 1);
224 
225  // Mark waves as updated
226  hasWaves_ = true;
227  }

229  /*
230  * Calculate fourier modes of charge density.
231  */
232  void MdEwaldPotential::computeKSpaceCharge()
233  {
234  // Compute waves if necessary
235  if (!hasWaves()) {
236  makeWaves();
237  }
238 
239  System::MoleculeIterator molIter;
240  Molecule::AtomIterator atomIter;
241  Vector rg; // scaled atom position vector
242  IntVector q; // wave indices
243  Vector qv; // wave vector
244  double dotqr; // dot product between q and r
245  double x, y; // real and imaginary parts of phasor
246  double charge; // atom charge
247  double TwoPi; // 2.0*pi
248  int i, j; // array index
249 
250  TwoPi = 2.0*Constants::Pi;
251 
252  // Clear rho for all waves
253  for (i = 0; i < rho_.size(); ++i) {
254  rho_[i] = DCMPLX(0.0, 0.0);
255  }
256 
257  // Loop over species, molecules atoms
258  int nSpecies = simulationPtr_->nSpecies();
259  for (int iSpecies = 0; iSpecies < nSpecies; ++iSpecies) {
260  systemPtr_->begin(iSpecies, molIter);
261  for ( ; molIter.notEnd(); ++molIter) {
262  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
263  boundaryPtr_->transformCartToGen(atomIter->position(), rg);
264  charge = (*atomTypesPtr_)[atomIter->typeId()].charge();
265 
266  // Loop over waves
267  for (i = 0; i < intWaves_.size(); ++i) {
268  q = intWaves_[i];
269  for (j = 0; j < Dimension; ++j) {
270  qv[j] = double(q[j]);
271  }
272  dotqr = rg.dot(qv)*TwoPi;
273  x = charge*cos(dotqr);
274  y = charge*sin(dotqr);
275  rho_[i] += DCMPLX(x, y);
276  }
277 
278  } // For atoms.
279  } // For molecules.
280  } // For species.
281 
282  }

284  /*
285  * Add k-space Coulomb forces for all atoms.
286  */
287  void MdEwaldPotential::addForces()
288  {
289  System::MoleculeIterator molIter, imolIter, jmolIter;
290  Molecule::AtomIterator atomIter, iatomIter, jatomIter;
291  Vector b[Dimension]; // array of reciprocal basis vectors
292  Vector rg; // scaled atom position vector
293  Vector fg; // generalized force on atom
294  Vector df; // force contribution
295  IntVector q; // wave index
296  double charge; // atom charge
297  double x, y; // Real and imaginary parts of phasor
298 
299  DCMPLX TwoPiIm; // 2.0*pi*I
300  DCMPLX de, expfactor;
301  double EPS(1.0E-10); // Tiny number to check if is charge
302  double volume = boundaryPtr_->volume();
303  double TwoPi = 2.0*Constants::Pi;
304  TwoPiIm = TwoPi * Constants::Im;
305  int nSpecies = simulationPtr_->nSpecies();
306  int type;
307  int i, j;
308 
309  // Compute Fourier components of charge density.
310  computeKSpaceCharge();
311 
312  // Store reciprocal lattice vectors
313  for (j = 0; j < Dimension; ++j) {
314  b[j] = boundaryPtr_->reciprocalBasisVector(j);
315  }
316 
317  // Loop over species, molecules atoms
318  for (int iSpecies = 0; iSpecies < nSpecies; ++iSpecies) {
319  systemPtr_->begin(iSpecies, molIter);
320  for ( ; molIter.notEnd(); ++molIter) {
321  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
322  type = atomIter->typeId();
323  charge = (*atomTypesPtr_)[type].charge();
324 
325  if (fabs(charge) > EPS) {
326  boundaryPtr_->transformCartToGen(atomIter->position(), rg);
327 
328  // Tabulate the exponential factors.
329  fexp0_[0] = exp(TwoPiIm * rg[0] * double(base0_));
330  de = exp(TwoPiIm * rg[0]);
331  for (i = 1; i < fexp0_.size(); ++i) {
332  fexp0_[i] = fexp0_[i-1] * de;
333  }
334 
335  fexp1_[0] = exp(TwoPiIm * rg[1] * double(base1_));
336  de = exp(TwoPiIm * rg[1]);
337  for (i = 1; i < fexp1_.size(); ++i) {
338  fexp1_[i] = fexp1_[i-1] * de;
339  }
340 
341  fexp2_[0] = exp(TwoPiIm * rg[2] * double(base2_));
342  de = exp(TwoPiIm * rg[2]);
343  for (i = 1; i < fexp2_.size(); ++i) {
344  fexp2_[i] = fexp2_[i-1] * de;
345  }
346 
347  // Accumulating forces.
348  fg.zero();
349  for (i = 0; i < intWaves_.size(); ++i) {
350  q = intWaves_[i];
351  for (j = 0; j < Dimension; ++j) {
352  df[j] = double(q[j]);
353  }
354  expfactor = fexp0_[q[0]-base0_] * fexp1_[q[1]-base1_] * fexp2_[q[2]-base2_];
355  x = expfactor.real();
356  y = expfactor.imag();
357  df *= g_[i]*( x*rho_[i].imag() - y*rho_[i].real() );
358  fg += df;
359  }
360  fg *= -2.0*charge/volume;
361 
362  // Transform to Cartesian coordinates
363  for (j = 0; j < Dimension; ++j) {
364  df.multiply(b[j], fg[j]);
365  atomIter->force() += df;
366  }
367  }
368 
369  } // for atoms
370  } // for molecules
371  } // for species
372  }

374  /*
375  * Calculate the k-space contribution to the Coulomb energy.
376  */
377  void MdEwaldPotential::computeEnergy()
378  {
379 
380  // Compute Fourier components of charge density.
381  computeKSpaceCharge();
382 
383  // Main loop over wavevectors
384  double x, y,rhoSq;
385  double energy = 0.0;
386  for (int i = 0; i < intWaves_.size(); ++i) {
387  x = rho_[i].real();
388  y = rho_[i].imag();
389  rhoSq = x*x + y*y;
390  energy += rhoSq*g_[i];
391  }
392  double volume = boundaryPtr_->volume();
393  energy /= volume;
394  // Note: A factor of 0.5 in the expression for the kspace energy
395  // kpart is cancelled our use of only half the wavevectors
396 
397  // Calculate self-energy correction to Ewald summation.
398  System::MoleculeIterator molIter;
399  Molecule::AtomIterator atomiter;
400  double charge;
401  double selfEnergy = 0.0;
402  int nSpecies = simulationPtr_->nSpecies();
403  int iAtomType;
404  for (int iSpecies = 0; iSpecies < nSpecies; ++iSpecies) {
405  systemPtr_->begin(iSpecies, molIter);
406  for ( ; molIter.notEnd(); ++molIter) {
407  for (molIter->begin(atomiter); atomiter.notEnd(); ++atomiter) {
408  iAtomType = atomiter->typeId();
409  charge = (*atomTypesPtr_)[iAtomType].charge();
410  selfEnergy += charge*charge;
411  }
412  }
413  }
414  double pi = Constants::Pi;
415  double alpha = ewaldInteraction_.alpha();
416  double epsilon = ewaldInteraction_.epsilon();
417  selfEnergy *= alpha/(4.0*sqrt(pi)*pi*epsilon);
418 
419  // Correct for conjugate wave contribution in k-part.
420  kSpaceEnergy_.set(energy - selfEnergy);
421  }

423  /*
424  * Compute the k contribution to stress.
425  * computeStress() for coulomb part in MdSystem.cpp is off.
426  */
427  void MdEwaldPotential::computeStress()
428  {
429  // Compute Fourier components of charge density.
430  computeKSpaceCharge();
431 
432  int i;
433  double x, y; //real and image part of rho[i].
434  Tensor K,stressTensor;// temp stress tensor.
435  IntVector q;//vector indices.
436  Vector qv,q0,q1,q2; //reciprocalVector.
437  Vector b0, b1, b2; // reciprocalBasisVector.
438  b0 = boundaryPtr_->reciprocalBasisVector(0);
439  b1 = boundaryPtr_->reciprocalBasisVector(1);
440  b2 = boundaryPtr_->reciprocalBasisVector(2);
441 
442  stressTensor.zero();
443  double alpha = ewaldInteraction_.alpha();
444  double ca = 0.25/(alpha*alpha);
445  for (i = 0; i < intWaves_.size(); ++i) {
446  q = intWaves_[i];
447  q0.multiply(b0, q[0]);
448  q1.multiply(b1, q[1]);
449  q2.multiply(b2, q[2]);
450  qv = q0;
451  qv += q1;
452  qv += q2;
453 
454  K.dyad(qv,qv);
455  K *= -2.0 * (ca + (1.0/ksq_[i]));
456  K.add(Tensor::Identity, K);
457  x = rho_[i].real();
458  y = rho_[i].imag();
459  K *= g_[i]*(x*x + y*y);
460  stressTensor += K;
461  }
462  double volume = boundaryPtr_->volume();
463  stressTensor /= volume*volume;
464  // Note: A factor of 0.5 in usual expression for stress is
465  // cancelled by our use of only half of the wavevectors.
466 
467  kSpaceStress_.set(stressTensor);
468  }
