130  void CompositionProfile::sample(long iStep)
131  {
132  Vector blengths;
133 
134  // Lengths of simulation box
135  blengths = system().boundary().lengths();
136 
137  if (isAtInterval(iStep)) {
138  Vector position;
139  double product;
140  System::ConstMoleculeIterator molIter;
141  Molecule::ConstAtomIterator atomIter;
142  int nSpecies, iSpecies, typeId;
143 
144  // Clear accumulators for the current timestep
145  for (int i = 0; i < nDirection_; ++i) {
146  for (int j = 0; j < nAtomType_; ++j){
147  currentAccumulators_[i+j*nDirection_].clear();
148  }
149  }
150 
151  // Select open mode for output files
152  std::ios_base::openmode mode = std::ios_base::out;
153  if (!isFirstStep_) {
154  mode = std::ios_base::out | std::ios_base::app;
155  }
156 
157  // Open log files
158  for (int i = 0; i < nDirection_; ++i) {
159  for (int j = 0; j < nAtomType_; ++j){
160  std::ostringstream oss;
161  oss << outputFileName();
162 
163  for (int k = 0; k < Dimension; ++k) {
164  oss << "_" << intVectors_[i][k];
165  }
166  oss << "_type" << j << ".log";
167 
168  fileMaster().openOutputFile(oss.str(),
169  logFiles_[i+j*nDirection_],
170  mode);
171  }
172  }
173 
174  // Loop over all atoms
175  nSpecies = system().simulation().nSpecies();
176  for (iSpecies = 0; iSpecies < nSpecies; ++iSpecies) {
177  system().begin(iSpecies, molIter);
178 
179  for ( ; molIter.notEnd(); ++molIter) {
180  molIter->begin(atomIter);
181 
182  for ( ; atomIter.notEnd(); ++atomIter) {
183  position = atomIter->position();
184  typeId = atomIter->typeId();
185  for (int i = 0; i < Dimension; ++i) {
186  position[i] /= blengths[i];
187  }
188  // Loop over direction vectors
189  for (int i = 0; i < nDirection_; ++i) {
190  product = position.dot(waveVectors_[i]);
191  product /= waveVectors_[i].abs();
192  accumulators_[i+typeId*nDirection_].sample(product);
193  currentAccumulators_[i+typeId*nDirection_].sample(product);
194  }
195 
196  }
197 
198  }
199  }
200 
201  ++nSample_;
202 
203  // Output to log files
204  for (int i = 0; i < nDirection_; ++i) {
205  for (int j = 0; j < nAtomType_; ++j){
206  currentAccumulators_[i+j*nDirection_].output(logFiles_[i+j*nDirection_]);
207  logFiles_[i+j*nDirection_] << std::endl;
208  }
209  }
210 
211  // Close log files
212  for (int i = 0; i < nDirection_; ++i) {
213  for (int j = 0; j < nAtomType_; ++j){
214  logFiles_[i+j*nDirection_].close();
215  }
216  }
217 
218  isFirstStep_ = false;
219  }
