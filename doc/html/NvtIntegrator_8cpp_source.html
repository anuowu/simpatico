110  /*
111  * Setup parameters before beginning of run.
112  */
113  void NvtIntegrator::setup()
114  {
115 
116  // Initialize state and clear statistics on first usage.
117  if (!isSetup()) {
118  clear();
119  setIsSetup();
120  }
121 
122  // Exchange atoms, build pair list, compute forces.
123  setupAtoms();
124 
125  // Calculate prefactors for acceleration
126  double dtHalf = 0.5*dt_;
127  double mass;
128  int nAtomType = prefactors_.capacity();
129  for (int i = 0; i < nAtomType; ++i) {
130  mass = simulation().atomType(i).mass();
131  prefactors_[i] = dtHalf/mass;
132  }
133 
134  // Initialize nAtom_, xiDot_
135  simulation().computeKineticEnergy();
136  #ifdef UTIL_MPI
137  atomStorage().computeNAtomTotal(domain().communicator());
138  #endif
139  if (domain().isMaster()) {
140  T_target_ = simulation().energyEnsemble().temperature();
141  nAtom_ = atomStorage().nAtomTotal();
142  T_kinetic_ = simulation().kineticEnergy()*2.0/double(3*nAtom_);
143  xiDot_ = (T_kinetic_/T_target_ -1.0)*nuT_*nuT_;
144  }
145  #ifdef UTIL_MPI
146  bcast(domain().communicator(), xiDot_, 0);
147  #endif
148 
149  }

151  void NvtIntegrator::integrateStep1()
152  {
153  Vector dv;
154  Vector dr;
155  double prefactor; // = 0.5*dt/mass
156  double dtHalf = 0.5*dt_;
157  double factor;
158  AtomIterator atomIter;
159 
160 
161  T_target_ = simulation().energyEnsemble().temperature();
162  factor = exp(-dtHalf*(xi_ + xiDot_*dtHalf));
163 
164  // 1st half of velocity Verlet.
165  atomStorage().begin(atomIter);
166  for ( ; atomIter.notEnd(); ++atomIter) {
167  atomIter->velocity() *= factor;
168  prefactor = prefactors_[atomIter->typeId()];
169  dv.multiply(atomIter->force(), prefactor);
170  atomIter->velocity() += dv;
171  dr.multiply(atomIter->velocity(), dt_);
172  atomIter->position() += dr;
173  }
174  }

176  void NvtIntegrator::integrateStep2()
177  {
178  Vector dv;
179  double prefactor; // = 0.5*dt/mass
180  double dtHalf = 0.5*dt_;
181  double factor;
182  AtomIterator atomIter;
183 
184  T_target_ = simulation().energyEnsemble().temperature();
185  factor = exp(-dtHalf*(xi_ + xiDot_*dtHalf));
186 
187  // 2nd half of velocity Verlet
188  atomStorage().begin(atomIter);
189  for ( ; atomIter.notEnd(); ++atomIter) {
190  prefactor = prefactors_[atomIter->typeId()];
191  dv.multiply(atomIter->force(), prefactor);
192  atomIter->velocity() += dv;
193  atomIter->velocity() *=factor;
194  }
195 
196  // Notify observers of change in velocity
197  simulation().velocitySignal().notify();
198 
199  // Update xiDot_ and xi_
200  simulation().computeKineticEnergy();
201  if (domain().isMaster()) {
202  xi_ += xiDot_*dtHalf;
203  T_kinetic_ = simulation().kineticEnergy()*2.0/double(3*nAtom_);
204  xiDot_ = (T_kinetic_/T_target_ - 1.0)*nuT_*nuT_;
205  xi_ += xiDot_*dtHalf;
206  }
207  #ifdef UTIL_MPI
208  bcast(domain().communicator(), xiDot_, 0);
209  bcast(domain().communicator(), xi_, 0);
210  #endif
211 
212  }
