33  /*
34  * Compute the bond tensor (Call on all processors).
35  */
36  void BondTensorAutoCorr::computeData()
37  {
38  MPI::Intracomm& communicator = simulation().domain().communicator();
39  BondStorage& storage = simulation().bondStorage();
40  Boundary& boundary = simulation().boundary();
41 
42  Tensor localTensor;
43  Vector dr;
44  double rsq;
45  GroupIterator<2> iter;
46  Atom* atom0Ptr;
47  Atom* atom1Ptr;
48  int isLocal0, isLocal1, i, j;
49 
50 
51  // Iterate over bonds
52  localTensor.zero();
53  storage.begin(iter);
54  for ( ; iter.notEnd(); ++iter) {
55  atom0Ptr = iter->atomPtr(0);
56  atom1Ptr = iter->atomPtr(1);
57  isLocal0 = !(atom0Ptr->isGhost());
58  isLocal1 = !(atom1Ptr->isGhost());
59  rsq = boundary.distanceSq(atom0Ptr->position(),
60  atom1Ptr->position(), dr);
61  if (rsq > 1.0E-6) {
62  dr /= sqrt(rsq);
63  assert(isLocal0 || isLocal1);
64  if (isLocal0 && isLocal1) {
65  // If both atoms are local
66  for (i = 0; i < Dimension; ++i) {
67  for (j = 0; j < Dimension; ++j) {
68  localTensor(i, j) += dr[i]*dr[j];
69  }
70  }
71  } else {
72  // If one atoms is local and one is a ghost
73  for (i = 0; i < Dimension; ++i) {
74  for (j = 0; j < Dimension; ++j) {
75  localTensor(i, j) += 0.5*dr[i]*dr[j];
76  }
77  }
78  }
79  }
80  }
81 
82  // Reduce partial sums from all processors, store on the master.
83  bondTensor_.zero();
84  communicator.Reduce(&localTensor(0,0), &bondTensor_(0,0),
85  Dimension*Dimension, MPI::DOUBLE, MPI::SUM, 0);
86 
87  if (communicator.Get_rank() != 0) {
88  bondTensor_.zero();
89  }
90  }

92  /*
93  * Return the current bond tensor value.
94  */
95  Tensor BondTensorAutoCorr::data()
96  {
97  // Remove trace
98  double trace = 0.0;
99  int i, j;
100  for (i = 0; i < Dimension; ++i) {
101  trace += bondTensor_(i,i);
102  }
103  trace = trace/double(Dimension);
104  for (i = 0; i < Dimension; ++i) {
105  bondTensor_(i,i) -= trace;
106  }
107 
108  // Scale traceless symmetric tensor
109  double factor = 1.0/sqrt(10.0*simulation().boundary().volume());
110  for (i = 0; i < Dimension; ++i) {
111  for (j = 0; j < Dimension; ++j) {
112  bondTensor_(i,j) *= factor;
113  }
114  }
115 
116  return bondTensor_;
117  }
