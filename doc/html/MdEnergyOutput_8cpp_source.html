52  /*
53  * Evaluate energy and print.
54  */
55  void MdEnergyOutput::sample(long iStep)
56  {
57  if (isAtInterval(iStep)) {
58  double potential = 0.0;
59  #ifndef INTER_NOPAIR
60  double pair = system().pairPotential().energy();
61  potential += pair;
62  outputFile_ << Dbl(pair);
63  #endif
64  #ifdef INTER_BOND
65  double bond = system().bondPotential().energy();
66  potential += bond;
67  outputFile_ << Dbl(bond);
68  #endif
69  #ifdef INTER_ANGLE
70  if (system().hasAnglePotential()) {
71  double angle = system().anglePotential().energy();
72  potential += angle;
73  outputFile_ << Dbl(angle);
74  }
75  #endif
76  #ifdef INTER_DIHEDRAL
77  if (system().hasDihedralPotential()) {
78  double dihedral = system().dihedralPotential().energy();
79  potential += dihedral;
80  outputFile_ << Dbl(dihedral);
81  }
82  #endif
83  #ifdef MCMD_LINK
84  if (system().hasLinkPotential()) {
85  double link = system().linkPotential().energy();
86  potential += link;
87  outputFile_ << Dbl(link);
88  }
89  #endif
90  #ifdef INTER_EXTERNAL
91  if (system().hasExternalPotential()) {
92  double external = system().externalPotential().energy();
93  potential += external;
94  outputFile_ << Dbl(external);
95  }
96  #endif
97  #ifdef INTER_TETHER
98  if (system().hasTetherPotential()) {
99  double tether = system().tetherPotential().energy();
100  potential += tether;
101  outputFile_ << Dbl(tether);
102  }
103  #endif
104  outputFile_ << Dbl(potential);
105  double kinetic = system().kineticEnergy();
106  outputFile_ << Dbl(kinetic);
107  double total = potential + kinetic;
108  outputFile_ << Dbl(total) << std::endl;
109  }
110  }
