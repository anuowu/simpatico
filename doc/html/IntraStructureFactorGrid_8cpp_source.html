101  g[1] = k;
102  for (l = 0; l <= k; ++l) {
103  g[2] = l;
104  starIds_[i] = j;
105  group.makeStar(g, star);

178  for (j = 0; j < nAtomTypeIdPair_; ++j) {
179  structureFactors_(i, j) = 0.0;

308  k = starIds_[i];
309  average = 0.0;

314  }
