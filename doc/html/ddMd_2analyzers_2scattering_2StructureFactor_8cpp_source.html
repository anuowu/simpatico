142  /*
143  * Increment structure factor.
144  */
145  void StructureFactor::sample(long iStep)
146  {
147  if (!isAtInterval(iStep)) {
148  UTIL_THROW("Time step index not a multiple of interval");
149  }
150  if (simulation().domain().isMaster()) {
151  std::ios_base::openmode mode = std::ios_base::out;
152  if (!isFirstStep_) {
153  mode = std::ios_base::out | std::ios_base::app;
154  }
155  std::string name = outputFileName("_max.dat");
156  simulation().fileMaster().openOutputFile(name, outputFile_, mode);
157  }
158 
159  isFirstStep_ = false;
160  Vector position;
161  std::complex<double> expFactor;
162  double product;
163  AtomIterator atomIter;
164  int i, j, typeId;
165 
166  makeWaveVectors();
167 
168  // Set all Fourier modes to zero
169  for (i = 0; i < nWave_; ++i) {
170  for (j = 0; j < nMode_; ++j) {
171  fourierModes_(i, j) = std::complex<double>(0.0, 0.0);
172  }
173  }
174 
175  simulation().atomStorage().begin(atomIter);
176  for ( ; atomIter.notEnd(); ++atomIter) {
177  position = atomIter->position();
178  typeId = atomIter->typeId();
179 
180  // Loop over wavevectors
181  for (i = 0; i < nWave_; ++i) {
182 
183  product = position.dot(waveVectors_[i]);
184  expFactor = exp( product*Constants::Im );
185  for (j = 0; j < nMode_; ++j) {
186  fourierModes_(i, j) += modes_(j, typeId)*expFactor;
187  }
188  }
189  }
190 
191  for (i = 0; i < nWave_; ++i) {
192  for (j = 0; j < nMode_; ++j) {
193  totalFourierModes_(i, j) = std::complex<double>(0.0, 0.0);
194  }
195  }
196 
197  #ifdef UTIL_MPI
198  // Loop over wavevectors
199  for (int i = 0; i < nWave_; ++i) {
200  for (int j = 0; j < nMode_; ++j) {
201  //Sum values from all processors.
202  simulation().domain().communicator().
203  Reduce(&fourierModes_(i, j), &totalFourierModes_(i, j),
204  1, MPI::DOUBLE_COMPLEX, MPI::SUM, 0);
205  }
206  }
207  #else
208  for (int i = 0; i < nWave_; ++i) {
209  for (int j = 0; j < nMode_; ++j) {
210  totalFourierModes_(i, j) = fourierModes_(i, j);
211  }
212  }
213  #endif
214 
215  if (simulation().domain().isMaster()) {
216  // Increment structure factors
217  double volume = simulation().boundary().volume();
218  double norm, maxValue, maxQ;
219  IntVector maxIntVector;
220  for (j = 0; j < nMode_; ++j) {
221  maxValue = 0.0;
222  maxQ = 0.0;
223  for (i = 1; i < nWave_; ++i) {
224  norm = std::norm(totalFourierModes_(i, j));
225  norm = norm/volume;
226  if ( norm >= maxValue ) {
227  maxValue = norm;
228  maxIntVector = waveIntVectors_[i];
229  maxQ = waveVectors_[i].abs();
230  }
231  structureFactors_(i, j) += norm;
232  }
233  outputFile_ << maxIntVector;
234  outputFile_ << Dbl(maxQ,20,8);
235  outputFile_ << Dbl(maxValue,20,8);
236  outputFile_ << std::endl;
237  }
238  outputFile_ << std::endl;
239  outputFile_.close();
240  }
241 
242  ++nSample_;
243 
244  }

247  * Calculate floating point wavevectors.
248  */
249  void StructureFactor::makeWaveVectors()
250  {
251  Vector dWave;
252  Boundary* boundaryPtr = &simulation().boundary();
253  int i, j;
254 
255  // Calculate wavevectors
256  for (i = 0; i < nWave_; ++i) {
257  waveVectors_[i] = Vector::Zero;
258  for (j = 0; j < Dimension; ++j) {
259  dWave = boundaryPtr->reciprocalBasisVector(j);
260  dWave *= waveIntVectors_[i][j];
261  waveVectors_[i] += dWave;
262  }
263  }
264  }
