48  void DdMdTrajectoryWriter::writeFrame(std::ofstream &file, long iStep)
49  {
50  BinaryFileOArchive ar(file);

52  if (domain().isMaster()) {
53 
54  ar << iStep;

57  Vector r;

73  if (r[j] >= 1.0) r[j] -= 1.0;
74  if (r[j] < 0.0) r[j] += 1.0;
