73  /*
74  * Generate, attempt and accept or reject a rigid molecule translation.
75  */
76  bool RigidDisplaceMove::move()
77  {
78  Vector oldPos, newPos, dr;
79  double newEnergy, oldEnergy;
80  Molecule* molPtr;
81  Atom* atomPtr;
82  int iAtom, j;
83 
84  incrementNAttempt();
85 
86  // Choose a molecule at random
87  molPtr = &(system().randomMolecule(speciesId_));
88 
89  // Calculate current energy and store old positions.
90  // Ignore covalent bonding energy, because it won't change.
91  oldEnergy = 0.0;
92  for (iAtom = 0; iAtom < nAtom_; ++iAtom) {
93  atomPtr = &molPtr->atom(iAtom);
94  oldPositions_[iAtom] = atomPtr->position();
95  #ifndef INTER_NOPAIR
96  oldEnergy += system().pairPotential().atomEnergy(*atomPtr);
97  #endif
98  #ifdef INTER_EXTERNAL
99  oldEnergy += system().externalPotential().atomEnergy(*atomPtr);
100  #endif
101  #ifdef INTER_TETHER
102  oldEnergy += system().atomTetherEnergy(*atomPtr);
103  #endif
104  }
105 
106  // Generate trial displacement Vector dr
107  for (j = 0; j < Dimension; ++j) {
108  dr[j] = random().uniform(-delta_, delta_);
109  }
110 
111  // Move every atom by dr and calculate new trial energy.
112  newEnergy = 0.0;
113  for (iAtom = 0; iAtom < nAtom_; ++iAtom) {
114  atomPtr = &molPtr->atom(iAtom);
115  atomPtr->position() += dr;
116  boundary().shift(atomPtr->position());
117  #ifndef INTER_NOPAIR
118  newEnergy += system().pairPotential().atomEnergy(*atomPtr);
119  #endif
120  #ifdef INTER_EXTERNAL
121  newEnergy += system().externalPotential().atomEnergy(*atomPtr);
122  #endif
123  #ifdef INTER_TETHER
124  newEnergy += system().atomTetherEnergy(*atomPtr);
125  #endif
126  }
127 
128  // Decide whether to accept the move
129  bool accept = random().metropolis(boltzmann(newEnergy - oldEnergy));
130 
131  if (accept) {
132 
133  #ifndef INTER_NOPAIR
134  // Update cell
135  for (iAtom = 0; iAtom < nAtom_; ++iAtom) {
136  system().pairPotential().updateAtomCell(molPtr->atom(iAtom));
137  }
138  #endif
139 
140  incrementNAccept();
141 
142  } else {
143 
144  // Return atoms to original positions
145  for (iAtom = 0; iAtom < nAtom_; ++iAtom) {
146  molPtr->atom(iAtom).position() = oldPositions_[iAtom];
147  }
148 
149  }
150  return accept;
151  }
