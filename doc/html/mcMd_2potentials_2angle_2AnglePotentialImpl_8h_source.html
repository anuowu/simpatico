348  double AnglePotentialImpl<Interaction>::atomEnergy(const Atom &atom) const
349  {
350  AtomAngleArray angles;
351  const Angle* anglePtr;
352  int iAngle;
353  Vector dr1; // R[1] - R[0]
354  Vector dr2; // R[2] - R[1]
355  double rsq1, rsq2, cosTheta;
356  double energy = 0.0;
357 
358  getAtomAngles(atom, angles);
359  for (iAngle = 0; iAngle < angles.size(); ++iAngle) {
360  anglePtr = angles[iAngle];
361  rsq1 = boundary().distanceSq(anglePtr->atom(1).position(),
362  anglePtr->atom(0).position(), dr1);
363  rsq2 = boundary().distanceSq(anglePtr->atom(2).position(),
364  anglePtr->atom(1).position(), dr2);
365  cosTheta = dr1.dot(dr2) / sqrt(rsq1 * rsq2);
366  energy += interaction().energy(cosTheta, anglePtr->typeId());
367  }
368 
369  return energy;

443  void AnglePotentialImpl<Interaction>::computeStressImpl(T& stress) const
444  {
445  Vector dr1, dr2, force1, force2;
446  const Atom *atom0Ptr, *atom1Ptr, *atom2Ptr;
447 
448  setToZero(stress);
449 
450  // Iterate over all angles in System.
451  System::ConstMoleculeIterator molIter;
452  Molecule::ConstAngleIterator angleIter;
453  for (int iSpec = 0; iSpec < simulation().nSpecies(); ++iSpec) {
454  if (simulation().species(iSpec).nAngle() > 0) {
455  for (begin(iSpec, molIter); molIter.notEnd(); ++molIter) {
456  for (molIter->begin(angleIter); angleIter.notEnd(); ++angleIter){
457  atom0Ptr = &(angleIter->atom(0));
458  atom1Ptr = &(angleIter->atom(1));
459  atom2Ptr = &(angleIter->atom(2));
460 
461  boundary().distanceSq(atom1Ptr->position(),
462  atom0Ptr->position(), dr1);
463  boundary().distanceSq(atom2Ptr->position(),
464  atom1Ptr->position(), dr2);
465 
466  // force1 -- along dr1; force2 -- along dr2.
467  interaction().force(dr1, dr2,
468  force1, force2, angleIter->typeId());
469 
470  dr1 *= -1.0;
471  dr2 *= -1.0;
472  incrementPairStress(force1, dr1, stress);
473  incrementPairStress(force2, dr2, stress);
474  }
475  }
476  }
477  }
478 
479  stress /= boundary().volume();
480  normalizeStress(stress);
481  }
