210  /*
211  * Calculate repulsive potential energy for particle of type i.
212  *
213  * 7.905694 = 25/sqrt(10), which is the negative of the minimum of the
214  * unshifted 9-3 potential.
215  */
216  inline double BoxExternal::energy(double d, int type) const
217  {
218  if (d < cutoff_) {
219  double r3i;
220  r3i = d * d * d;
221  if (r3i < 0.129615*sigmaCb_) r3i = 0.129615 * sigmaCb_;
222  r3i = sigmaCb_ / r3i;
223  return coeff_ * (r3i*(r3i*r3i - 7.5) + 7.905694);
224  } else {
225  return 0.0;
226  }
227  }

232  inline double BoxExternal::energy(const Vector& position, int type) const
233  {
234  double d, totalEnergy;

236  halfL *= 0.5;
237  totalEnergy = 0.0;
238  for (int i = 0; i < Dimension; ++i) {
239  d = position[i];
240  if (d > halfL[i]) d = halfL[i] + halfL[i] - d;
241  totalEnergy += energy(d, type);
242  }
