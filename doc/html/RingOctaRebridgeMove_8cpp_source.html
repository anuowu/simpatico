84  /*
85  * Generate, attempt and accept or reject a Monte Carlo move.
86  *
87  * A complete move involve the following stepts:
88  *
89  * 1) choose a molecule-i randomly;
90  * 2) find molecule-j of the same type as i, and monomer pairs satisfying
91  * the distance criterion exist;
92  *
93  */
94  bool RingOctaRebridgeMove::move()
95  {
96  Molecule *molPtr1; // pointer to the rebridging molecule
97  Molecule *molPtr2; // pointer to the rebridging molecule
98  Atom *mPtr, *nPtr; // atom pointers
99  Vector swapV;
100  int nAtom, im, in, molId2;
101  double energy;
102  bool found, accept;
103 
104  incrementNAttempt();
105 
106  // Randomly pick up a molecule.
107  molPtr1 = &(system().randomMolecule(speciesId_));
108  nAtom = molPtr1->nAtom();
109 
110  // Randomly choose the a atom in the tetra-group.
111  im = random().uniformInt(0, nAtom);
112 
113  // Search the rebriding sites.
114  found = scanBridge(molPtr1, im, molId2, in);
115  if (!found) return false;
116 
117  // Calcuate the energy change.
118  mPtr = &(molPtr1->atom(im));
119  molPtr2 = &(system().molecule(speciesId_, molId2));
120  nPtr = &(molPtr2->atom(in));
121 
122  energy = 0.0;
123  energy -= system().atomPotentialEnergy(*mPtr);
124  energy -= system().atomPotentialEnergy(*nPtr);
125 
126  // Swap positions of atoms m and n.
127  swapV = mPtr->position();
128 
129  //system().moveAtom(*mPtr, nPtr->position());
130  mPtr->position() = nPtr->position();
131  #ifndef SIMP_NOPAIR
132  system().pairPotential().updateAtomCell(*mPtr);
133  #endif
134 
135  //system().moveAtom(*nPtr, swapV);
136  nPtr->position() = swapV;
137  #ifndef SIMP_NOPAIR
138  system().pairPotential().updateAtomCell(*nPtr);
139  #endif
140 
141  energy += system().atomPotentialEnergy(*mPtr);
142  energy += system().atomPotentialEnergy(*nPtr);
143 
144  // Decide whether to accept or reject.
145  accept = random().metropolis(boltzmann(energy));
146 
147  if (accept) {
148  // Increment counter for accepted moves of this class.
149  incrementNAccept();
150  } else {
151  // Swap back bead positions.
152  swapV = mPtr->position();
153 
154  //system().moveAtom(*mPtr, nPtr->position());
155  mPtr->position() = nPtr->position();
156  #ifndef SIMP_NOPAIR
157  system().pairPotential().updateAtomCell(*mPtr);
158  #endif
159 
160  //system().moveAtom(*nPtr, swapV);
161  nPtr->position() = swapV;
162  #ifndef SIMP_NOPAIR
163  system().pairPotential().updateAtomCell(*nPtr);
164  #endif
165  }
166 
167  return accept;
168  }

170  /*
171  * Return true if the octahedron formed by the 6 closely approaching atoms
172  * satisfy the 8 distance criterion.
173  */
174  bool RingOctaRebridgeMove::
175  scanBridge(Molecule* molPtr1, int im, int& molId2, int &in)
176  {
177  Molecule *molPtr2;
178  Atom *pn;
179  Vector aPos, bPos, cPos, dPos, mPos, nPos;
180  int nAtom, ia, ib, ic, id, dmn, dnm;
181  int nNeighbor, j, *idList, *molIdList;
182  int molId1, choice, nGroup;
183  double lenSq, minSq, maxSq;
184  bool validIndex, found;
185 
186  // Number of atoms per molecule.
187  nAtom = molPtr1->nAtom();
188 
189  // The a and b Atom id.
190  ia = modId(im-1, nAtom);
191  ib = modId(im+1, nAtom);
192 
193  aPos = molPtr1->atom(ia).position();
194  bPos = molPtr1->atom(ib).position();
195  mPos = molPtr1->atom(im).position();
196 
197  // Prepare for the scan.
198  found = false;
199  maxSq = upperBridge_ * upperBridge_;
200  minSq = lowerBridge_ * lowerBridge_;
201 
202  // Check the am bond.
203  lenSq = boundary().distanceSq(aPos, mPos);
204  if (lenSq >= maxSq || lenSq <= minSq) {
205  in = -1; // invalid value
206  molId2 = -1; // invalid value
207  return found;
208  }
209 
210  // Check the mb bond.
211  lenSq = boundary().distanceSq(mPos, bPos);
212  if (lenSq >= maxSq || lenSq <= minSq) {
213  in = -1; // invalid value
214  molId2 = -1; // invalid value
215  return found;
216  }
217 
218  // Get the neighbor list.
219  #ifndef SIMP_NOPAIR
220  system().pairPotential().cellList().getNeighbors(mPos, neighbors_);
221  nNeighbor = neighbors_.size();
222  idList = new int[nNeighbor];
223  molIdList = new int[nNeighbor];
224 
225  // Loop over neighboring monomers and identify the octa-group.
226  nGroup = 0;
227  molId1 = molPtr1->id();
228  for (j = 0; j < nNeighbor; ++j) {
229  pn = neighbors_[j];
230  in = pn->indexInMolecule();
231  molId2 = pn->molecule().id();
232 
233  validIndex = true;
234 
235  if (molId2 == molId1) {
236  // Check if im and in are separated by at least two atoms.
237  dmn = in - im;
238  if (dmn < 0) dmn += nAtom;
239 
240  dnm = im - in;
241  if (dnm < 0) dnm += nAtom;
242 
243  if (dmn <= 2 && dnm <= 2) validIndex = false;
244  }
245 
246  if (pn->molecule().species().id() != speciesId_)
247  validIndex = false;
248 
249  if (validIndex) {
250 
251  molPtr2 = &(system().molecule(speciesId_, molId2));
252 
253  ic = modId(in-1, nAtom);
254  id = modId(in+1, nAtom);
255 
256  cPos = molPtr2->atom(ic).position();
257  dPos = molPtr2->atom(id).position();
258  nPos = molPtr2->atom(in).position();
259 
260  // Distance test.
261  lenSq = boundary().distanceSq(cPos, mPos);
262  if (lenSq > minSq && lenSq < maxSq) {
263 
264  lenSq = boundary().distanceSq(mPos, dPos);
265  if (lenSq > minSq && lenSq < maxSq) {
266 
267  lenSq = boundary().distanceSq(aPos, nPos);
268  if (lenSq > minSq && lenSq < maxSq) {
269 
270  lenSq = boundary().distanceSq(nPos, bPos);
271  if (lenSq > minSq && lenSq < maxSq) {
272 
273  lenSq = boundary().distanceSq(cPos, nPos);
274  if (lenSq > minSq && lenSq < maxSq) {
275 
276  lenSq = boundary().distanceSq(nPos, dPos);
277  if (lenSq > minSq && lenSq < maxSq) {
278 
279  idList[nGroup] = in;
280  molIdList[nGroup] = molId2;
281  ++nGroup;
282 
283  } } } } } } // 6 additional distance test
284 
285  } // if the atom id "in" is valid
286 
287  } // loop over neighbors
288  #endif
289 
290  if (nGroup > 0) {
291  found = true;
292  choice = random().uniformInt(0,nGroup);
293  in = idList[choice];
294  molId2 = molIdList[choice];
295  } else {
296  found = false;
297  in = -1; // Invalid value.
298  molId2 = -1; // Invalid value.
299  }
300 
301  delete [] molIdList;
302  delete [] idList;
303  return found;
304 
305  }
