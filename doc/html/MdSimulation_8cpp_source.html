309  * Read and implement commands in an input script.
310  */
311  void MdSimulation::readCommands(std::istream &in)
312  {
313  std::string command;
314  #ifndef UTIL_MPI
315  std::istream& inBuffer = in;
316  #else
317  std::stringstream inBuffer;
318  std::string line;
319  #endif
320 
321  bool readNext = true;
322  while (readNext) {
323 
324  #ifdef UTIL_MPI
325  if (!hasIoCommunicator() || isIoProcessor()) {
326  getNextLine(in, line);
327  }
328  if (hasIoCommunicator()) {
329  bcast<std::string>(communicator(), line, 0);
330  }
331  inBuffer.clear();
332  for (unsigned i=0; i < line.size(); ++i) {
333  inBuffer.put(line[i]);
334  }
335  #endif
336 
337  inBuffer >> command;
338  Log::file().setf(std::ios_base::left);
339  Log::file().width(15);
340  Log::file() << command;
341 
342  if (isRestarting_) {
343 
344  if (command == "RESTART") {
345  int endStep;
346  inBuffer >> endStep;
347  Log::file() << " " << iStep_ << " to "
348  << endStep << std::endl;
349  simulate(endStep, isRestarting_);
350  isRestarting_ = false;
351  } else {
352  UTIL_THROW("Missing RESTART command");
353  }
354 
355  } else {
356 
357  if (command == "FINISH") {
358  Log::file() << std::endl;
359  readNext = false;
360  } else {
361  bool success;
362  success = readCommand(command, inBuffer);
363  if (!success) {
364  Log::file() << "Error: Unknown command " << std::endl;
365  readNext = false;
366  }
367  }
368 
369  }
370  }

384  bool MdSimulation::readCommand(std::string const & command,
385  std::istream& in)
386  {
387  std::string filename;
388  std::ifstream inputFile;
389  std::ofstream outputFile;
390  bool success = true;
391 
392  if (command == "READ_CONFIG") {
393  in >> filename;
394  Log::file() << Str(filename, 15) << std::endl;
395  fileMaster().openInputFile(filename, inputFile);
396  std::cout << "Opened config file" << std::endl;
397  system().readConfig(inputFile);
398  std::cout << "Finished reading config file" << std::endl;
399  inputFile.close();
400  } else
401  if (command == "THERMALIZE") {
402  double temperature;
403  in >> temperature;
404  Log::file() << Dbl(temperature, 15, 6) << std::endl;
405  system().setBoltzmannVelocities(temperature);
406  system().removeDriftVelocity();
407  } else
408  if (command == "SIMULATE") {
409  int endStep;
410  in >> endStep;
411  Log::file() << Int(endStep, 15) << std::endl;
412  bool isContinuation = false;
413  simulate(endStep, isContinuation);
414  } else
415  if (command == "CONTINUE") {
416  if (iStep_ == 0) {
417  UTIL_THROW("Attempt to continue simulation when iStep_ == 0");
418  }
419  int endStep;
420  in >> endStep;
421  Log::file() << Int(endStep, 15) << std::endl;
422  bool isContinuation = true;
423  simulate(endStep, isContinuation);
424  } else
425  if (command == "ANALYZE_CONFIGS") {
426  int min, max;
427  in >> min >> max >> filename;
428  Log::file() << Int(min, 15) << Int(max, 15)
429  << Str(filename, 20) << std::endl;
430  analyzeConfigs(min, max, filename);
431  } else
432  if (command == "WRITE_CONFIG") {
433  in >> filename;
434  Log::file() << Str(filename, 15) << std::endl;
435  fileMaster().openOutputFile(filename, outputFile);
436  system().writeConfig(outputFile);
437  outputFile.close();
438  } else
439  if (command == "WRITE_PARAM") {
440  in >> filename;
441  Log::file() << Str(filename, 15) << std::endl;
442  fileMaster().openOutputFile(filename, outputFile);
443  writeParam(outputFile);
444  outputFile.close();
445  } else
446  if (command == "SET_CONFIG_IO") {
447  std::string classname;
448  in >> classname;
449  Log::file() << Str(classname, 15) << std::endl;
450  system().setConfigIo(classname);
451  } else
452  if (command == "ANALYZE_TRAJECTORY") {
453  std::string classname;
454  std::string filename;
455  int min, max;
456  in >> min >> max >> classname >> filename;
457  Log::file() << " " << Str(classname,15)
458  << " " << Str(filename, 15)
459  << std::endl;
460  analyzeTrajectory(min, max, classname, filename);
461  } else
462  if (command == "GENERATE_MOLECULES") {
463  DArray<double> diameters;
464  DArray<int> capacities;
465  diameters.allocate(nAtomType());
466  capacities.allocate(nSpecies());
467 
468  // Parse command
469  in >> system().boundary();
470  Log::file() << " " << system().boundary();
471  Label capacityLabel("Capacities:");
472  in >> capacityLabel;
473  for (int iSpecies = 0; iSpecies < nSpecies(); ++iSpecies) {
474  in >> capacities[iSpecies];
475  Log::file() << " " << capacities[iSpecies];
476  }
477  Label diameterLabel("Diameters:");
478  in >> diameterLabel;
479  for (int iType=0; iType < nAtomType(); iType++) {
480  in >> diameters[iType];
481  Log::file() << " " << diameters[iType];
482  }
483  Log::file() << std::endl;
484 
485  system().generateMolecules(capacities, diameters);
486  } else
487  #ifndef UTIL_MPI
488  #ifndef SIMP_NOPAIR
489  if (command == "SET_PAIR") {
490  std::string paramName;
491  int typeId1, typeId2;
492  double value;
493  in >> paramName >> typeId1 >> typeId2 >> value;
494  Log::file() << " " << paramName
495  << " " << typeId1 << " " << typeId2
496  << " " << value << std::endl;
497  system().pairPotential()
498  .set(paramName, typeId1, typeId2, value);
499  } else
500  #endif
501  #ifdef SIMP_BOND
502  if (command == "SET_BOND") {
503  std::string paramName;
504  int typeId;
505  double value;
506  in >> paramName >> typeId >> value;
507  Log::file() << " " << paramName << " " << typeId
508  << " " << value << std::endl;
509  system().bondPotential().set(paramName, typeId, value);
510  } else
511  #endif
512  #ifdef SIMP_ANGLE
513  if (command == "SET_ANGLE") {
514  std::string paramName;
515  int typeId;
516  double value;
517  in >> paramName >> typeId >> value;
518  Log::file() << " " << paramName << " " << typeId
519  << " " << value << std::endl;
520  system().anglePotential().set(paramName, typeId, value);
521  } else
522  #endif
523  #ifdef SIMP_DIHEDRAL
524  if (command == "SET_DIHEDRAL") {
525  std::string paramName;
526  int typeId;
527  double value;
528  in >> paramName >> typeId >> value;
529  Log::file() << " " << paramName << " " << typeId
530  << " " << value << std::endl;
531  system().dihedralPotential().set(paramName, typeId, value);
532  } else
533  #endif
534  #endif
535  {
536  success = false;
537  }
538  return success;

541  /*
542  * Run a simulation.
543  */
544  void MdSimulation::simulate(int endStep, bool isContinuation)
545  {
546  Timer timer;
547 
548  if (isContinuation) {
549  Log::file() << "Continuing simulation from iStep = "
550  << iStep_ << std::endl;
551  } else {
552  iStep_ = 0;
553  system().shiftAtoms();
554  #ifdef UTIL_COULOMB
555  if (system().hasCoulombPotential()) {
556  system().coulombPotential().makeWaves();
557  }
558  #endif
559  system().calculateForces();
560  analyzerManager().setup();
561  system_.mdIntegrator().setup();
562  }
563  int beginStep = iStep_;
564  int nStep = endStep - beginStep;
565 
566  #ifdef SIMP_NOPAIR
567  // When the pair potential is disabled, require that
568  // Analyzer::baseInterval > 0 to guarantee periodic
569  // shifting of atomic positions into primary cell.
570  UTIL_CHECK(Analyzer::baseInterval > 0);
571  #endif
572 
573  // Main loop
574  Log::file() << std::endl;
575  timer.start();
576  for ( ; iStep_ < endStep; ++iStep_) {
577 
578  // Sample scheduled analyzers
579  if (Analyzer::baseInterval > 0) {
580  if (iStep_ % Analyzer::baseInterval == 0) {
581  system().shiftAtoms();
582  analyzerManager().sample(iStep_);
583  }
584  }
585 
586  if (saveInterval_ > 0) {
587  if (iStep_ % saveInterval_ == 0) {
588  save(saveFileName_);
589  }
590  }
591 
592  // Take one MD step with the MdIntegrator
593  system_.mdIntegrator().step();
594  }
595  timer.stop();
596  double time = timer.time();
597  double rstep = double(nStep);
598 
599  // Shift final atomic positions
600  system().shiftAtoms();
601 
602  // Final analyzers and restart
603  if (Analyzer::baseInterval > 0) {
604  if (iStep_ % Analyzer::baseInterval == 0) {
605  analyzerManager().sample(iStep_);
606  }
607  }
608  if (saveInterval_ > 0) {
609  if (iStep_ % saveInterval_ == 0) {
610  save(saveFileName_);
611  }
612  }
613 
614  // Final analyzer output
615  analyzerManager().output();
616 
617  // Output timing results
618  Log::file() << std::endl;
619  Log::file() << "Time Statistics" << std::endl;
620  Log::file() << "endStep " << endStep << std::endl;
621  Log::file() << "nStep " << nStep << std::endl;
622  Log::file() << "run time "
623  << time << " sec" << std::endl;
624  Log::file() << "time / nStep "
625  << time / rstep << " sec" << std::endl;
626  Log::file() << "time / (nStep*nAtom) "
627  << time / (rstep*double(system().nAtom()))
628  << " sec" << std::endl;
629  Log::file() << std::endl;
630  Log::file() << std::endl;
631 
632  #ifndef SIMP_NOPAIR
633  Log::file() << "PairList Statistics" << std::endl;
634  Log::file() << "maxNPair "
635  << system().pairPotential().pairList().maxNPair()
636  << std::endl;
637  Log::file() << "maxNAtom "
638  << system().pairPotential().pairList().maxNAtom()
639  << std::endl;
640  int buildCounter = system().pairPotential().pairList().buildCounter();
641  Log::file() << "buildCounter "
642  << buildCounter
643  << std::endl;
644  Log::file() << "steps / build "
645  << rstep/double(buildCounter)
646  << std::endl;
647  Log::file() << std::endl;
648  #endif
649 
650  }

656  MdSimulation::analyzeConfigs(int min, int max, std::string basename)
657  {
658  // Preconditions
659  UTIL_CHECK(min >= 0);
660  UTIL_CHECK(max > min);
661  UTIL_CHECK(Analyzer::baseInterval);
662  UTIL_CHECK(analyzerManager().size() > 0);
663 
664  Timer timer;
665  std::string filename;
666  std::stringstream indexString;
667  std::ifstream configFile;
668  int nConfig;
669  nConfig = max - min + 1;
670 
671  // Main loop
672  Log::file() << "begin main loop" << std::endl;
673  timer.start();
674  for (iStep_ = min; iStep_ <= max; ++iStep_) {
675 
676  indexString << iStep_;
677  filename = basename;
678  filename += indexString.str();
679  fileMaster().openInputFile(filename, configFile);
680 
681  // Clear the stringstream
682  indexString.str("");
683 
684  system().readConfig(configFile);
685  configFile.close();
686 
687  #ifndef SIMP_NOPAIR
688  // Build the system CellList
689  system().pairPotential().buildPairList();
690  #endif
691 
692  #ifndef NDEBUG
693  isValid();
694  #endif
695 
696  // Initialize analyzers (taking in molecular information).
697  if (iStep_ == min) analyzerManager().setup();
698 
699  // Sample property values
700  analyzerManager().sample(iStep_);
701 
702  // Clear out the System for the next readConfig.
703  system().removeAllMolecules();
704 
705  }
706  timer.stop();
707  Log::file()<< "end main loop" << std::endl;
708 
709  // Output results of all analyzers to output files
710  analyzerManager().output();
711 
712  // Output time
713  Log::file() << std::endl;
714  Log::file() << "nConfig " << nConfig << std::endl;
715  Log::file() << "run time " << timer.time()
716  << " sec" << std::endl;
717  Log::file() << "time / config " << timer.time()/double(nConfig)
718  << " sec" << std::endl;
719  Log::file() << std::endl;
720 
721  }

723  /*
724  * Open, read and analyze a trajectory file
725  */
726  void MdSimulation::analyzeTrajectory(int min, int max,
727  std::string classname,
728  std::string filename)
729  {
730  // Preconditions
731  UTIL_CHECK(min >= 0);
732  UTIL_CHECK(max > min);
733  UTIL_CHECK(Analyzer::baseInterval);
734  UTIL_CHECK(analyzerManager().size() > 0);
735 
736  // Construct TrajectoryReader
737  TrajectoryReader* trajectoryReaderPtr;
738  trajectoryReaderPtr
739  = system().trajectoryReaderFactory().factory(classname);
740  if (!trajectoryReaderPtr) {
741  std::string message;
742  message = "Invalid TrajectoryReader class name " + classname;
743  UTIL_THROW(message.c_str());
744  }
745 
746  // Open trajectory file
747  Log::file() << "Reading " << filename << std::endl;
748  trajectoryReaderPtr->open(filename);
749 
750  // Main loop over trajectory frames
751  Timer timer;
752  Log::file() << "Begin main loop" << std::endl;
753  bool hasFrame = true;
754  timer.start();
755  for (iStep_ = 0; iStep_ <= max && hasFrame; ++iStep_) {
756  hasFrame = trajectoryReaderPtr->readFrame();
757  if (hasFrame) {
758  #ifndef SIMP_NOPAIR
759  // Build the system PairList
760  system().pairPotential().buildPairList();
761  #endif
762  #ifdef UTIL_DEBUG
763  isValid();
764  #endif
765  // Initialize analyzers (taking in molecular information).
766  if (iStep_ == min) analyzerManager().setup();
767  // Sample property values only for iStep >= min
