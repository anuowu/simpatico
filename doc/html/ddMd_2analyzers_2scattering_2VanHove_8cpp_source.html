144  void VanHove::sample(long iStep)
145  {
146  if (isAtInterval(iStep)) {
147 
148  Vector position;
149  std::complex<double> expFactor;
150  double product, coeff;
151  AtomIterator atomIter;
152  int i, typeId;
153 
154  makeWaveVectors();
155 
156  // Set all Fourier modes to zero
157  for (i = 0; i < nWave_; ++i) {
158  fourierModes_[i] = std::complex<double>(0.0, 0.0);
159  }
160 
161  simulation().atomStorage().begin(atomIter);
162  for ( ; atomIter.notEnd(); ++atomIter) {
163  position = atomIter->position();
164  typeId = atomIter->typeId();
165  coeff = atomTypeCoeffs_[typeId];
166 
167  // Loop over wavevectors
168  for (i = 0; i < nWave_; ++i) {
169 
170  product = position.dot(waveVectors_[i]);
171  expFactor = exp( product*Constants::Im );
172  fourierModes_[i] += coeff*expFactor;
173  }
174  }
175 
176  for (i = 0; i < nWave_; ++i) {
177  totalFourierModes_[i] = std::complex<double>(0.0, 0.0);
178  }
179 
180  #ifdef UTIL_MPI
181  // Loop over wavevectors
182  for (int i = 0; i < nWave_; ++i) {
183  //Sum values from all processors.
184  simulation().domain().communicator().
185  Reduce(&fourierModes_[i], &totalFourierModes_[i],
186  1, MPI::DOUBLE_COMPLEX, MPI::SUM, 0);
187  }
188  #else
189  for (int i = 0; i < nWave_; ++i) {
190  totalFourierModes_[i] = fourierModes_[i];
191  }
192  #endif
193 
194  if (simulation().domain().isMaster()) {
195  // Add Fourier modes to autocorrelation accumulators
196  double sqrtV = sqrt(simulation().boundary().volume());
197  for (int i = 0; i < nWave_; ++i) {
198  accumulators_[i].sample(totalFourierModes_[i]/sqrtV);
199  }
200  }
201  ++nSample_;
202  }
203 
204  }

209  void VanHove::makeWaveVectors()
210  {
211  Vector dWave;
212  Boundary* boundaryPtr = &simulation().boundary();
213  int i, j;
214 
215  // Calculate wavevectors
216  for (i = 0; i < nWave_; ++i) {
217  waveVectors_[i] = Vector::Zero;
218  for (j = 0; j < Dimension; ++j) {
219  dWave = boundaryPtr->reciprocalBasisVector(j);
220  dWave *= waveIntVectors_[i][j];
221  waveVectors_[i] += dWave;
222  }
223  }
