81  void VelProf::sample(long iStep)
82  {
83  if (isAtInterval(iStep)) {
84  int i, iSpec;
85  System::MoleculeIterator molIter;
86  Molecule::AtomIterator atomIter;
87  double Lz = system().boundary().lengths()[2];
88  double zcoord;
89  int imed=int(Lz/(2.0*slabWidth_));
90  double vx, maxvx=-10.0, minvx=10.0, deltaP;
91  Atom* atomPtrmax=0;
92  Atom* atomPtrmin=0;
93 
94  for (i=0; i < nSlabs_; ++i){
95  velx_[i] = 0.0;
96  nAtomi_[i] = 0;
97  }
98 
99  // Loop over all atoms
100  for (iSpec = 0; iSpec < nSpec_; ++iSpec) {
101  for (system().begin(iSpec, molIter); molIter.notEnd(); ++molIter) {
102  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
103  zcoord = (atomIter->position())[2];
104  i = int(zcoord/slabWidth_);
105  if(zcoord == Lz) i = nSlabs_-1;
106  vx = (atomIter->velocity())[0];
107  velx_[i] += vx;
108  nAtomi_[i] += 1;
109  if(iStep > 0){
110  if (i == 0) {
111  if (vx < minvx){
112  minvx = vx;
113  atomPtrmin = &(*atomIter);
114  }
115  }
116  if (i == imed){
117  if (vx > maxvx){
118  maxvx = vx;
119  atomPtrmax = &(*atomIter);
120  }
121  }
122  }
123  }
124  }
125  }
126 
127  // Calculate mean velocity for each slab.
128  if(iStep > 0){
129  iBlock_ += 1;
130 
131  for (i=0; i < nSlabs_; ++i){
132  if (nAtomi_[i] != 0 ){
133  velx_[i] = velx_[i]/(double)nAtomi_[i];
134  }
135  else {
136  velx_[i] = 0.0;
137  }
138  accumulator_[i].sample(velx_[i], outputFile_);
139  }
140 
141  if (iBlock_ == nSamplePerBlock_) {
142  outputFile_ << std::endl;
143  iBlock_ = 0;
144  }
145 
146  if(atomPtrmin!=0 && atomPtrmax!=0){
147  (atomPtrmin->velocity())[0] = maxvx;
148  (atomPtrmax->velocity())[0] = minvx;
149  deltaP = maxvx-minvx;
150  totalMomentum_ += deltaP;
151  }
152  }
153 
154  } // if isAtInterval
