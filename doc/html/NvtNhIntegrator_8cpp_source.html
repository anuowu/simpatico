108  void NvtNhIntegrator::setup()
109  {
110  double mass, dtHalf;
111  int nAtomType = simulation().nAtomType();
112  int nAtom = system().nAtom();
113 
114  T_kinetic_ = system().kineticEnergy()*2.0/double(3*nAtom);
115  T_target_ = energyEnsemblePtr_->temperature();
116  xiDot_ = (T_kinetic_/T_target_ -1.0)*nuT_*nuT_;
117  xi_ = 0.0;
118 
119  dtHalf = 0.5*dt_;
120  for (int i = 0; i < nAtomType; ++i) {
121  mass = simulation().atomType(i).mass();
122  prefactors_[i] = dtHalf/mass;
123  }
124  }

126  /*
127  * Nose-Hoover integrator step.
128  *
129  * This implements a reversible Velocity-Verlet MD NVT integrator step.
130  *
131  * Reference: Winkler, Kraus, and Reineker, J. Chem. Phys. 102, 9018 (1995).
132  */
133  void NvtNhIntegrator::step()
134  {
135  Vector dv;
136  Vector dr;
137  System::MoleculeIterator molIter;
138  double dtHalf = 0.5*dt_;
139  double prefactor;
140  double factor;
141  Molecule::AtomIterator atomIter;
142  int iSpecies, nSpecies;
143  int nAtom;
144 
145  T_target_ = energyEnsemblePtr_->temperature();
146  nSpecies = simulation().nSpecies();
147  nAtom = system().nAtom();
148 
149  factor = exp(-dtHalf*(xi_ + xiDot_*dtHalf));
150 
151  // 1st half velocity Verlet, loop over atoms
152  for (iSpecies = 0; iSpecies < nSpecies; ++iSpecies) {
153  system().begin(iSpecies, molIter);
154  for ( ; molIter.notEnd(); ++molIter) {
155 
156  molIter->begin(atomIter);
157  for ( ; atomIter.notEnd(); ++atomIter) {
158 
159  atomIter->velocity() *= factor;
160 
161  prefactor = prefactors_[atomIter->typeId()];
162  dv.multiply(atomIter->force(), prefactor);
163  //dv.multiply(atomIter->force(), dtHalf);
164 
165  atomIter->velocity() += dv;
166  dr.multiply(atomIter->velocity(), dt_);
167 
168  atomIter->position() += dr;
169 
170  }
171 
172  }
173  }
174 
175  // First half of update of xi_
176  xi_ += xiDot_*dtHalf;
177 
178  #ifndef INTER_NOPAIR
179  // Rebuild the pair list if necessary
180  if (!system().pairPotential().isPairListCurrent()) {
181  system().pairPotential().buildPairList();
182  }
183  #endif
184 
185  system().calculateForces();
186 
187  // 2nd half velocity Verlet, loop over atoms
188  for (iSpecies=0; iSpecies < nSpecies; ++iSpecies) {
189  system().begin(iSpecies, molIter);
190  for ( ; molIter.notEnd(); ++molIter) {
191  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
192  prefactor = prefactors_[atomIter->typeId()];
193  dv.multiply(atomIter->force(), prefactor);
194  atomIter->velocity() += dv;
195  atomIter->velocity() *=factor;
196  }
197  }
198  }
199 
200  // Update xiDot and complete update of xi_
201  T_kinetic_ = system().kineticEnergy()*2.0/double(3*nAtom);
202  xiDot_ = (T_kinetic_/T_target_ -1.0)*nuT_*nuT_;
203  xi_ += xiDot_*dtHalf;
204 
205  }
206 
207 }
