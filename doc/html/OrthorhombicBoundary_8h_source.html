505  * Calculate squared distance by minimum image convention.
506  */
507  inline
508  double OrthorhombicBoundary::distanceSq(const Vector &r1, const Vector &r2,
509  IntVector& translate) const
510  {
511  double dr;
512  double norm = 0.0;
513  for (int i = 0; i < Dimension; ++i) {
514  dr = r1[i] - r2[i];
515  if ( fabs(dr) > halfLengths_[i] ) {
516  if (dr > 0.0) {
517  dr -= lengths_[i];
518  translate[i] = -1;
519  } else {
520  dr += lengths_[i];
521  translate[i] = +1;
522  }
523  assert(fabs(dr) <= halfLengths_[i]);
524  } else {
525  translate[i] = 0;
526  }
527  norm += dr*dr;
528  }
529  return norm;
