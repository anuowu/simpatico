596  /*
597  * Set velocities to Boltzmannian random values, for all atoms in System.
598  */
599  void MdSystem::setBoltzmannVelocities(double temperature)
600  {
601  Simulation& sim = simulation();
602  Random &random = sim.random();
603  double scale;
604  double mass;
605  MoleculeIterator molIter;
606  int nSpec = sim.nSpecies();
607  int iSpec, j;
608 
609  Molecule::AtomIterator atomIter;
610  for (iSpec = 0; iSpec < nSpec; ++iSpec) {
611  for (begin(iSpec, molIter); molIter.notEnd(); ++molIter) {
612  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
613  mass = sim.atomType(atomIter->typeId()).mass();
614  scale = sqrt(temperature/mass);
615  for (j = 0; j < Dimension; ++j) {
616  atomIter->velocity()[j] = scale*random.gaussian();
617  }
618  }
619  }
620  }

624  /*
625  * Subtract average velocity from all atomic velocities.
626  */
627  Vector MdSystem::removeDriftVelocity()
628  {
629  Vector average(0.0);
630  MoleculeIterator molIter;
631  Molecule::AtomIterator atomIter;
632  int nSpec = simulation().nSpecies();
633  int nAtom = 0;
634  int iSpec;
635 
636  for (iSpec = 0; iSpec < nSpec; ++iSpec) {
637  for (begin(iSpec, molIter); molIter.notEnd(); ++molIter) {
638  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
639  average += atomIter->velocity();
640  ++nAtom;
641  }
642  }
643  }
644  average /= double(nAtom);
645  for (iSpec = 0; iSpec < nSpec; ++iSpec) {
646  for (begin(iSpec, molIter); molIter.notEnd(); ++molIter) {
647  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
648  atomIter->velocity() -= average;
649  }
650  }
651  }
652 
653  return average;
654  }

741  /*
742  * Return total kinetic energy.
743  */
744  double MdSystem::kineticEnergy() const
745  {
746  const Simulation& sim = simulation();
747  double KE, mass;
748  ConstMoleculeIterator molIter;
749  Molecule::ConstAtomIterator atomIter;
750  int iSpec, typeId;
751 
752  KE = 0.0;
753  for (iSpec=0; iSpec < sim.nSpecies(); ++iSpec) {
754  for (begin(iSpec, molIter); molIter.notEnd(); ++molIter) {
755  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
756  typeId = atomIter->typeId();
757  mass = sim.atomType(typeId).mass();
758  KE += atomIter->velocity().square()*mass;
759  }
760  }
761  }
762 
763  return 0.5*KE;
764  }

771  template <typename T>
772  void MdSystem::computeKineticStressImpl(T& stress) const
773  {
774  Vector p;
775  double mass;
776  ConstMoleculeIterator molIter;
777  Molecule::ConstAtomIterator atomIter;
778  const Vector* vPtr;
779  const Simulation& sim = simulation();
780 
781  setToZero(stress);
782  for (int iSpec=0; iSpec < sim.nSpecies(); ++iSpec) {
783  for (begin(iSpec, molIter); molIter.notEnd(); ++molIter) {
784  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
785  vPtr = &atomIter->velocity();
786  mass = sim.atomType(atomIter->typeId()).mass();
787  p.multiply(*vPtr, mass);
788  incrementPairStress(*vPtr, p, stress);
789  }
790  }
791  }
792  stress /= boundary().volume();
793  normalizeStress(stress);
794  }
