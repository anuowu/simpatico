111  void Atom::unpackAtom(Buffer& buffer)

114  buffer.unpack<int>(i);

117  setTypeId(i);
118  buffer.unpack<Vector>(position());
119  buffer.unpack<Vector>(velocity());
120  unsigned int ui;
121  buffer.unpack<unsigned int>(ui);
122  plan().setFlags(ui);
123  buffer.unpack<unsigned int>(ui);
124  groups() = ui;
125  if (hasAtomContext_) {
126  buffer.unpack<AtomContext>(context());
127  }
128 
129  // Unpack Mask
130  Mask& m = mask();
131  m.clear();
132  int size;
133  buffer.unpack<int>(size);
134  for (int j = 0; j < size; ++j) {
135  buffer.unpack<int>(i);
136  m.append(i);
137  }
