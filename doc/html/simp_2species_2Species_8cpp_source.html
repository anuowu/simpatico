77  */
78  void Species::readSpeciesParam(std::istream &in)
79  {
80  read<int>(in, "nAtom", nAtom_);

82  read<int>(in, "nBond", nBond_);
83  #endif

91 
92  readDArray<int>(in, "atomTypeIds", atomTypeIds_, nAtom_);
93 
94  #ifdef SIMP_BOND

96  readDArray<SpeciesBond>(in, "speciesBonds", speciesBonds_,
97  nBond_);
98  // Make atomBondIdArrays
99  int bondId, atomId1, atomId2;
100  for (bondId = 0; bondId < nBond_; ++bondId) {
101  atomId1 = speciesBond(bondId).atomId(0);
102  atomId2 = speciesBond(bondId).atomId(1);
103  atomBondIdArrays_[atomId1].append(bondId);
104  atomBondIdArrays_[atomId2].append(bondId);
105  }
106  }
107  #endif
108 
109  #ifdef SIMP_ANGLE
110  if (nAngle_ > 0) {
111  readDArray<SpeciesAngle>(in, "speciesAngles", speciesAngles_,
112  nAngle_);
113  // Make atomAngleIdArrays
114  int angleId, atomId1, atomId2, atomId3;
115  for (angleId = 0; angleId < nAngle_; ++angleId) {
116  atomId1 = speciesAngle(angleId).atomId(0);
117  atomId2 = speciesAngle(angleId).atomId(1);
118  atomId3 = speciesAngle(angleId).atomId(2);
119  atomAngleIdArrays_[atomId1].append(angleId);
120  atomAngleIdArrays_[atomId2].append(angleId);
121  atomAngleIdArrays_[atomId3].append(angleId);
122  }
123  }
124  #endif
125 
126  #ifdef SIMP_DIHEDRAL
127  // Make atomDihedralIdArrays
128  if (nDihedral_ > 0) {
129  readDArray<SpeciesDihedral>(in, "speciesDihedrals", speciesDihedrals_,
130  nDihedral_);
131  int dihedralId;
132  int tAtomId1, tAtomId2, tAtomId3, tAtomId4;
133  for (dihedralId = 0; dihedralId < nDihedral_; ++dihedralId) {
134  tAtomId1 = speciesDihedral(dihedralId).atomId(0);
135  tAtomId2 = speciesDihedral(dihedralId).atomId(1);
136  tAtomId3 = speciesDihedral(dihedralId).atomId(2);
137  tAtomId4 = speciesDihedral(dihedralId).atomId(3);
138  atomDihedralIdArrays_[tAtomId1].append(dihedralId);
139  atomDihedralIdArrays_[tAtomId2].append(dihedralId);
140  atomDihedralIdArrays_[tAtomId3].append(dihedralId);
141  atomDihedralIdArrays_[tAtomId4].append(dihedralId);
142  }
143  }

404  /*
405  * Check validity of Species. Return true if valid, or throw an Exception.
406  */
407  bool Species::isValid() const
408  {
409 
410  if (atomTypeIds_.isAllocated()) {
411 
412  // Check atomTypeIds array
413  if (!isMutable()) {
414  for (int i = 0; i < nAtom_; ++i) {
415  if (atomTypeIds_[i] < 0) {
416  UTIL_THROW("Negative AtomTypeIds_ in non-mutable Species");
417  }
418  }
419  }
420 
421  #ifdef SIMP_BOND
422  {
423  // Loop over all bonds (if any) in speciesBonds_ array
424  int atomId, bondId, atomId0, atomId1, j;
425  bool hasBond;
426  for (bondId = 0; bondId < nBond_; ++bondId) {
427 
428  atomId0 = speciesBond(bondId).atomId(0);
429  atomId1 = speciesBond(bondId).atomId(1);
430 
431  // Check that atomIds are in valid range and unequal
432  if (atomId0 < 0 || atomId0 >= nAtom_) {
433  UTIL_THROW("Invalid atomId0 in speciesBonds_");
434  }
435  if (atomId1 < 0 || atomId1 >= nAtom_) {
436  UTIL_THROW("Invalid atomId1 in speciesBonds_");
437  }
438  if (atomId0 == atomId1) {
439  UTIL_THROW("Equal atom ids in a SpeciesBond");
440  }
441 
442  // Check that bondId is in atombondIdArrays_ for atomId0
443  hasBond = false;
444  for (j = 0; j < atomBondIdArrays_[atomId0].size(); ++j) {
445  if (atomBondIdArrays_[atomId0][j] == bondId) {
446  hasBond = true;
447  break;
448  }
449  }
450  if (!hasBond) {
451  UTIL_THROW("BondId missing from atomBondIdArrays_");
452  }
453 
454  // Check that bondId is in atombondIdArrays_ for atomId1
455  hasBond = false;
456  for (j = 0; j < atomBondIdArrays_[atomId1].size(); ++j) {
457  if (atomBondIdArrays_[atomId1][j] == bondId) {
458  hasBond = true;
459  break;
460  }
461  }
462  if (!hasBond) {
463  UTIL_THROW("BondId missing from atomBondIdArrays_");
464  }
465 
466  }
467 
468  // Loop over atomBondIdArrays for all atoms
469  int bondCounter = 0;
470  for (atomId = 0; atomId < nAtom_; ++atomId) {
471  for (j = 0; j < atomBondIdArrays_[atomId].size(); ++j) {
472  bondId = atomBondIdArrays_[atomId][j];
473 
474  if (bondId < 0 || bondId >= nBond_)
475  UTIL_THROW("Bond index out of range");
476 
477  atomId0 = speciesBond(bondId).atomId(0);
478  atomId1 = speciesBond(bondId).atomId(1);
479  if (atomId != atomId0 && atomId != atomId1)
480  UTIL_THROW("Inconsistency in atomBondId and bond");
481  ++bondCounter;
482 
483  }
484  }
485  if (bondCounter != 2*nBond_) {
486  UTIL_THROW("Inconsistency in total number of bonds");
487  }
488  }
489  #endif
490 
491  #ifdef SIMP_ANGLE
492  {
493  // Loop over all angles (if any) in speciesAngles_ array
494  int angleId, id, id2, atomId, atomId2(-1), j;
495  bool hasAngles;
496  for (angleId = 0; angleId < nAngle_; ++angleId) {
497 
498  for (id = 0; id < 3; ++id) {
499  atomId = speciesAngle(angleId).atomId(id);
500 
501  // Check that atomIds are in valid range and unequal
502  if (atomId < 0 || atomId >= nAtom_) {
503  UTIL_THROW("Invalid atomId in speciesAngles_");
504  }
505  for (id2 = id + 1; id2 < 3; ++id2) {
506  atomId2 = speciesAngle(angleId).atomId(id2);
507  if (atomId2 == atomId) {
508  UTIL_THROW("Equal atom ids in a SpeciesAngle");
509  }
510  }
511 
512  // Check that angleId is in atomAngleIdArrays_ for atomId
513  hasAngles = false;
514  for (j = 0; j < atomAngleIdArrays_[atomId].size(); ++j) {
515  if (atomAngleIdArrays_[atomId][j] == angleId) {
516  hasAngles = true;
517  break;
518  }
519  }
520  if (!hasAngles) {
521  UTIL_THROW("AngleId missing from atomAngleIdArrays_");
522  }
523  }
524 
525  }
526 
527  // Loop over atomAngleIdArrays for all atoms
528  int angleCounter = 0;
529  bool hasAtom;
530  for (atomId = 0; atomId < nAtom_; ++atomId) {
531  for (j = 0; j < atomAngleIdArrays_[atomId].size(); ++j) {
532  angleId = atomAngleIdArrays_[atomId][j];
533  if (angleId < 0 || angleId >= nAngle_)
534  UTIL_THROW("Angle index out of range");
535 
536  hasAtom = false;
537  for (id = 0; id < 3; ++id) {
538  atomId2 = speciesAngle(angleId).atomId(id);
539  if (atomId2 == atomId) {
540  hasAtom = true;
541  break;
542  }
543  }
544  if (!hasAtom) {
545  UTIL_THROW("Inconsistency in atomAngleId and angle");
546  }
547  ++angleCounter;
548  }
549  }
550  if (angleCounter != 3*nAngle_) {
551  UTIL_THROW("Inconsistency in total number of angles");
552  }
553  }
554  #endif
555 
556  #ifdef SIMP_DIHEDRAL
557  {
558  // Loop over all dihedrals (if any) in speciesDihedrals_ array
559  int dihedralId, tId, tId2, tAtomId, tAtomId2, j;
560  bool hasDihedral;
561 
562  for (dihedralId = 0; dihedralId < nDihedral_; ++dihedralId) {
563 
564  for (tId = 0; tId < 4; ++tId) {
565  tAtomId = speciesDihedral(dihedralId).atomId(tId);
566 
567  // Check that atomIds are in valid range and unequal
568  if (tAtomId < 0 || tAtomId >= nAtom_) {
569  UTIL_THROW("Invalid atomId in speciesDihedral_");
570  }
571  for (tId2 = tId + 1; tId2 < 4; ++tId2) {
572  tAtomId2 = speciesDihedral(dihedralId).atomId(tId2);
573  if (tAtomId2 == tAtomId) {
574  UTIL_THROW("Equal atom ids in a SpeciesDihedral");
575  }
576  }
577 
578  // Check that dihedralId is in atomDihedralIdArrays_ for tAtomId
579  hasDihedral = false;
580  for (j = 0; j < atomDihedralIdArrays_[tAtomId].size(); ++j) {
581  if (atomDihedralIdArrays_[tAtomId][j] == dihedralId) {
582  hasDihedral = true;
583  break;
584  }
585  }
586  if (!hasDihedral) {
587  UTIL_THROW("DihedralId missing from atomDihedralIdArrays_");
588  }
589  }
590 
591  }
592 
593  // Loop over atomDihedralIdArrays for all atoms.
594  int dihedralCounter = 0;
595  bool tHasAtom;
596  for (tAtomId = 0; tAtomId < nAtom_; ++tAtomId) {
597  for (j = 0; j < atomDihedralIdArrays_[tAtomId].size(); ++j) {
598  dihedralId = atomDihedralIdArrays_[tAtomId][j];
599  if (dihedralId < 0 || dihedralId >= nDihedral_)
600  UTIL_THROW("Dihedral index out of range");
601 
602  tHasAtom = false;
603  for (tId = 0; tId < 4; ++tId) {
604  tAtomId2 = speciesDihedral(dihedralId).atomId(tId);
605  if (tAtomId2 == tAtomId) {
606  tHasAtom = true;
607  break;
608  }
609  }
610  if (!tHasAtom) {
611  UTIL_THROW("Inconsistency in atomDihedralId and dihedral");
612  }
613  ++dihedralCounter;
614  }
615  }
616  if (dihedralCounter != 4*nDihedral_) {
617  UTIL_THROW("Inconsistency in total number of dihedrals");
618  }
619  }
620  #endif
621 
622  } else {
623 
624  if (moleculeCapacity_ != 0) {
625  UTIL_THROW("Species not allocated but moleculeCapacity != 0");
626  }
627 
628  }
629 
630  return true;
631  }
