92  /*
93  * Generate, attempt and accept or reject a Hybrid MD/MC move.
94  */
95  bool HybridNphMdMove::move()
96  {
97  System::MoleculeIterator molIter;
98  Molecule::AtomIterator atomIter;
99  double oldEnergy, newEnergy;
100  int iSpec;
101  int nSpec = simulation().nSpecies();
102 
103  bool accept;
104 
105  if (nphIntegratorPtr_ == NULL) {
106  UTIL_THROW("null integrator pointer");
107  }
108  // Increment counter for attempted moves
109  incrementNAttempt();
110 
111  // Store old boundary lengths.
112  Vector oldLengths = system().boundary().lengths();
113 
114  // Store old atom positions in oldPositions_ array.
115  for (iSpec = 0; iSpec < nSpec; ++iSpec) {
116  mdSystemPtr_->begin(iSpec, molIter);
117  for ( ; molIter.notEnd(); ++molIter) {
118  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
119  oldPositions_[atomIter->id()] = atomIter->position();
120  }
121  }
122  }
123 
124  // Initialize MdSystem
125  #ifndef SIMP_NOPAIR
126  mdSystemPtr_->pairPotential().buildPairList();
127  #endif
128  mdSystemPtr_->calculateForces();
129  mdSystemPtr_->setBoltzmannVelocities(energyEnsemble().temperature());
130  nphIntegratorPtr_->setup();
131 
132  // generate integrator variables from a Gaussian distribution
133  Random& random = simulation().random();
134 
135  double temp = system().energyEnsemble().temperature();
136 
137  double volume = system().boundary().volume();
138 
139  if (mode_ == Cubic) {
140  // one degree of freedom
141  // barostat_energy = 1/2 (1/W) eta_x^2
142  double sigma = sqrt(temp/barostatMass_);
143  nphIntegratorPtr_->setEta(0, sigma*random.gaussian());
144  } else if (mode_ == Tetragonal) {
145  // two degrees of freedom
146  // barostat_energy = 1/2 (1/W) eta_x^2 + 1/2 (1/(2W)) eta_y^2
147  double sigma1 = sqrt(temp/barostatMass_);
148  nphIntegratorPtr_->setEta(0, sigma1*random.gaussian());
149  double sigma2 = sqrt(temp/barostatMass_/2.0);
150  nphIntegratorPtr_->setEta(1, sigma2*random.gaussian());
151  } else if (mode_ == Orthorhombic) {
152  // three degrees of freedom
153  // barostat_energy = 1/2 (1/W) (eta_x^2 + eta_y^2 + eta_z^2)
154  double sigma = sqrt(temp/barostatMass_);
155  nphIntegratorPtr_->setEta(0, sigma*random.gaussian());
156  nphIntegratorPtr_->setEta(1, sigma*random.gaussian());
157  nphIntegratorPtr_->setEta(2, sigma*random.gaussian());
158  }
159 
160  // Store old energy
161  oldEnergy = mdSystemPtr_->potentialEnergy();
162  oldEnergy += mdSystemPtr_->kineticEnergy();
163  oldEnergy += system().boundaryEnsemble().pressure()*volume;
164  oldEnergy += nphIntegratorPtr_->barostatEnergy();
165 
166  // Run a short MD simulation
167  for (int iStep = 0; iStep < nStep_; ++iStep) {
168  nphIntegratorPtr_->step();
169  }
170 
171  volume = system().boundary().volume();
172 
173  // Calculate new energy
174  newEnergy = mdSystemPtr_->potentialEnergy();
175  newEnergy += mdSystemPtr_->kineticEnergy();
176  newEnergy += system().boundaryEnsemble().pressure()*volume;
177  newEnergy += nphIntegratorPtr_->barostatEnergy();
178 
179  // Decide whether to accept or reject
180  accept = random.metropolis( boltzmann(newEnergy-oldEnergy) );
181 
182  // Accept move
183  if (accept) {
184 
185  #ifndef SIMP_NOPAIR
186  // Rebuild the McSystem cellList using the new positions.
187  system().pairPotential().buildCellList();
188  #endif
189 
190  // Increment counter for the number of accepted moves.
191  incrementNAccept();
192 
193  } else {
194 
195  // Restore old boundary lengths
196  system().boundary().setOrthorhombic(oldLengths);
197 
198  // Restore old atom positions
199  for (iSpec = 0; iSpec < nSpec; ++iSpec) {
200  mdSystemPtr_->begin(iSpec, molIter);
201  for ( ; molIter.notEnd(); ++molIter) {
202  molIter->begin(atomIter);
203  for ( ; atomIter.notEnd(); ++atomIter) {
204  atomIter->position() = oldPositions_[atomIter->id()];
205  }
206  }
207  }
208 
209  }
210 
211  return accept;
212 
213  }
