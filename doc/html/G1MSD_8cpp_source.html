100  /*
101  * Evaluate end-to-end vectors of all chains, add to ensemble.
102  */
103  void G1MSD::sample(long iStep)
104  {
105  if (!isAtInterval(iStep)) return;
106 
107  // Confirm that nMolecule has remained constant, and nMolecule > 0.
108  if (nMolecule_ <= 0) {
109  UTIL_THROW("nMolecule <= 0");
110  }
111  if (nMolecule_ != system().nMolecule(speciesId_)) {
112  UTIL_THROW("Number of molecules has changed.");
113  }
114 
115  Vector r;
116  IntVector shift;
117  Molecule* moleculePtr;
118  Vector lengths = system().boundary().lengths();
119  int i, j, k;
120  int iatom = 0;
121  for (i = 0; i < nMolecule_; ++i) {
122  moleculePtr = &system().molecule(speciesId_, i);
123  for (j = 0 ; j < nAtom_; j++) {
124  r = moleculePtr->atom(j).position();
125  system().boundary().shift(r);
126 
127  // Compare current r to previous position, oldPositions_[iatom]
128  system().boundary().distanceSq(r, oldPositions_[iatom], shift);
129 
130  // If this atom crossed a boundary, increment its shift vector
131  shifts_[iatom] += shift;
132 
133  // Reconstruct true position
134  for (k = 0; k < Dimension; ++k) {
135  truePositions_[iatom][k] = r[k] + shifts_[iatom][k]*lengths[k];
136  }
137 
138  // Store current position in box for comparison to next one
139  oldPositions_[iatom] = r;
140 
141  iatom++;
142  }
143  }
144  accumulator_.sample(truePositions_);
145 
146  }
