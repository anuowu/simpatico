138  void AtomMSD::setup()
139  {
140  if (!isInitialized_) {
141  UTIL_THROW("Error: object is not initialized");
142  }
143 
144  // Set number of molecules of this species in the System.
145  nMolecule_ = system().nMolecule(speciesId_);
146  accumulator_.setNEnsemble(nMolecule_);
147  accumulator_.clear();
148 
149  // Store initial positions, and set initial shift vectors.
150  Vector r;
151  IntVector zero(0);
152  for (int i = 0; i < nMolecule_; ++i) {
153  r = system().molecule(speciesId_, i).atom(atomId_).position();
154  system().boundary().shift(r);
155  oldPositions_[i] = r;
156  shifts_[i] = zero;

160  /*
161  * Evaluate end-to-end vectors of all chains, add to ensemble.
162  */
163  void AtomMSD::sample(long iStep)
164  {
165  if (!isAtInterval(iStep)) return;
166 
167  // Confirm that nMolecule has remained constant, and nMolecule > 0.
168  if (nMolecule_ <= 0) {
169  UTIL_THROW("nMolecule <= 0");
170  }
171  if (nMolecule_ != system().nMolecule(speciesId_)) {
172  UTIL_THROW("Number of molecules has changed.");
173  }
174 
175  Vector r;
176  IntVector shift;
177  Vector lengths = system().boundary().lengths();
178  int i, j;
179  for (i = 0; i < nMolecule_; ++i) {
180 
181  r = system().molecule(speciesId_, i).atom(atomId_).position();
182  system().boundary().shift(r);
183 
184  // Compare current r to previous position, oldPositions_[i]
185  system().boundary().distanceSq(r, oldPositions_[i], shift);
186 
187  // If this atom crossed a boundary, increment its shift vector
188  shifts_[i] += shift;
189 
190  // Reconstruct true position
191  for (j = 0; j < Dimension; ++j) {
192  truePositions_[i][j] = r[j] + shifts_[i][j]*lengths[j];
193  }
194 
195  // Store current position in box for comparison to next one
196  oldPositions_[i] = r;
197 
198  }
199  accumulator_.sample(truePositions_);
200 
201  }
