118  /*
119  * Increment cos factor.
120  */
121  void OrderParamNucleation::sample(long iStep)
122  {
123  if (isAtInterval(iStep)) {
124 
125  Vector blengths;
126  Boundary* boundaryPtr = &simulation().boundary();
127  blengths = boundaryPtr->lengths();
128 
129  Vector position;
130  double slabLength, product, cosFactor;
131  AtomIterator atomIter;
132  int typeId, bin;
133 
134  simulation().atomStorage().begin(atomIter);
135  for ( ; atomIter.notEnd(); ++atomIter) {
136  position = atomIter->position();
137  typeId = atomIter->typeId();
138 
139  slabLength = blengths[parallelIndex_]/nBin_;
140  bin = position[parallelIndex_]/slabLength;
141 
142  product = position[perpIndex_]*2.0*Constants::Pi*periodicity_;
143  product /= blengths[perpIndex_];
144  cosFactor = cos(product)*cos(product);
145  cosFactors_(typeId, bin) += cosFactor;
146  }
147 
148  #ifdef UTIL_MPI
149  // Loop over wavevectors
150  for (int i = 0; i < nAtomType_; ++i) {
151  for (int j = 0; j < nBin_; ++j) {
152  //Sum values from all processors.
153  simulation().domain().communicator().
154  Reduce(&cosFactors_(i, j), &totalCosFactors_(i, j),
155  1, MPI::DOUBLE, MPI::SUM, 0);
156  }
157  }
158  #else
159  for (int i = 0; i < nAtomType_; ++i) {
160  for (int j = 0; j < nBin_; ++j) {
161  totalCosFactors_(i, j) = cosFactors_(i, j);
162  }
163  }
164  #endif
165 
166  ++nSample_;
167  }
168 
169  }
