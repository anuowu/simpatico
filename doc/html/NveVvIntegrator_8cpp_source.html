80  /*
81  * Verlet MD NVE integrator step
82  *
83  * This method implements the algorithm:
84  *
85  * vm(n) = v(n) + 0.5*a(n)*dt
86  * x(n+1) = x(n) + vm(n)*dt
87  *
88  * calculate acceleration a(n+1)
89  *
90  * v(n+1) = vm(n) + 0.5*a(n+1)*dt
91  *
92  * where x is position, v is velocity, and a is acceleration.
93  */
94  void NveVvIntegrator::step()
95  {
96  Vector dv;
97  Vector dr;
98  System::MoleculeIterator molIter;
99  double prefactor;
100  int iSpecies, nSpecies;
101 
102  nSpecies = simulation().nSpecies();
103 
104  // 1st half velocity Verlet, loop over atoms
105  #if USE_ITERATOR
106  Molecule::AtomIterator atomIter;
107  for (iSpecies=0; iSpecies < nSpecies; ++iSpecies) {
108  system().begin(iSpecies, molIter);
109  for ( ; molIter.notEnd(); ++molIter) {
110  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
111 
112  prefactor = prefactors_[atomIter->typeId()];
113 
114  dv.multiply(atomIter->force(), prefactor);
115  atomIter->velocity() += dv;
116 
117  dr.multiply(atomIter->velocity(), dt_);
118  atomIter->position() += dr;
119 
120  }
121  }
122  }
123  #else
124  Atom* atomPtr;
125  int ia;
126  for (iSpecies=0; iSpecies < nSpecies; ++iSpecies) {
127  system().begin(iSpecies, molIter);
128  for ( ; molIter.notEnd(); ++molIter) {
129  for (ia=0; ia < molIter->nAtom(); ++ia) {
130  atomPtr = &molIter->atom(ia);
131  prefactor = prefactors_[atomPtr->typeId()];
132 
133  dv.multiply(atomPtr->force(), prefactor);
134  atomPtr->velocity() += dv;
135  dr.multiply(atomPtr->velocity(), dt_);
136 
137  atomPtr->position() += dr;
138 
139  }
140  }
141  }
142  #endif
143 
144  system().calculateForces();
145 
146  // 2nd half velocity Verlet, loop over atoms
147  #if USE_ITERATOR
148  for (iSpecies=0; iSpecies < nSpecies; ++iSpecies) {
149  system().begin(iSpecies, molIter);
150  for ( ; molIter.notEnd(); ++molIter) {
151  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
152  prefactor = prefactors_[atomIter->typeId()];
153  dv.multiply(atomIter->force(), prefactor);
154  atomIter->velocity() += dv;
155  }
156  }
157  }
158  #else
159  for (iSpecies=0; iSpecies < nSpecies; ++iSpecies) {
160  system().begin(iSpecies, molIter);
161  for ( ; molIter.notEnd(); ++molIter)
162  {
163  for (ia=0; ia < molIter->nAtom(); ++ia) {
164  atomPtr = &molIter->atom(ia);
165  prefactor = prefactors_[atomPtr->typeId()];
166  dv.multiply(atomPtr->force(), prefactor);
167  atomPtr->velocity() += dv;
168  }
169  }
170  }
171  #endif
172 
173  #ifndef INTER_NOPAIR
174  if (!system().pairPotential().isPairListCurrent()) {
175  system().pairPotential().buildPairList();
176  }
177  #endif
178 
179  }
180 
181 }
