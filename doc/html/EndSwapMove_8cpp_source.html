117  /*
118  * Generate, attempt and accept or reject a Monte Carlo move.
119  */
120  bool EndSwapMove::move()
121  {
122  double newEnergy, oldEnergy;
123  Molecule* molPtr;
124  Atom* atomPtr;
125  int i, nAtom;
126 
127  incrementNAttempt();
128 
129  molPtr = &(system().randomMolecule(speciesId_));
130  nAtom = molPtr->nAtom();
131 
132  // Calculate old molecule energy = pair + external
133  oldEnergy = 0.0;
134  #ifndef INTER_NOPAIR
135  oldEnergy += system().pairPotential().moleculeEnergy(*molPtr);
136  #endif
137  #ifdef INTER_EXTERNAL
138  for (i = 0; i < nAtom; ++i) {
139  atomPtr = &molPtr->atom(i);
140  oldEnergy += system().externalPotential().atomEnergy(*atomPtr);
141  }
142  #endif
143 
144  // Reverse sequence of atom types
145  for (i = 0; i < nAtom; ++i) {
146  assert(molPtr->atom(i).typeId() == atomTypeIds_[i]);
147  molPtr->atom(i).setTypeId(atomTypeIds_[nAtom - 1 - i]);
148  }
149 
150  // Calculate new energy (with reversed sequence of atom types).
151  newEnergy = 0.0;
152  #ifndef INTER_NOPAIR
153  newEnergy += system().pairPotential().moleculeEnergy(*molPtr);
154  #endif
155  #ifdef INTER_EXTERNAL
156  for (i = 0; i < nAtom; ++i) {
157  molPtr->atom(i).setTypeId(atomTypeIds_[nAtom - 1 - i]);
158  atomPtr = &molPtr->atom(i);
159  newEnergy += system().externalPotential().atomEnergy(*atomPtr);
160  }
161  #endif
162 
163  // Restore original sequence of atom type Ids
164  for (i = 0; i < nAtom; ++i) {
165  molPtr->atom(i).setTypeId(atomTypeIds_[i]);
166  }
167 
168  // Decide whether to accept or reject
169  bool accept = random().metropolis(boltzmann(newEnergy-oldEnergy));
170 
171  if (accept) {
172 
173  // Store all atomic positions
174  for (i = 0; i < nAtom; ++i) {
175  positions_[i] = molPtr->atom(i).position();
176  }
177 
178  // Reverse sequence of atomic positions
179  for (i = 0; i < nAtom; ++i) {
180  atomPtr = &molPtr->atom(i);
181  atomPtr->position() = positions_[nAtom - 1 - i];
182  #ifndef INTER_NOPAIR
183  system().pairPotential().updateAtomCell(*atomPtr);
184  #endif
185  }
186 
187  incrementNAccept();
188 
189  }
190 
191  return accept;
192  }
