46  void McMdConfigIo::read(std::istream &in)
47  {
48 
49  // Read and set dimensions of simulation Boundary.
50  in >> Label("BOUNDARY");
51  in >> boundary() ;
52 
53  // Get molecules and read atomic positions
54  Label speciesLabel("species");
55  Label nMoleculeLabel("nMolecule");
56  Label moleculeLabel("molecule");
57  Species* speciesPtr;
58  Molecule* molPtr;
59  Molecule::AtomIterator atomIter;
60  int iSpecies, iSpeciesIn, nMolecule, iMol, iMolIn;
61 
62  in >> Label("MOLECULES");
63  for (iSpecies = 0; iSpecies < simulation().nSpecies(); ++iSpecies) {
64  speciesPtr = &simulation().species(iSpecies);
65  in >> Label("species") >> iSpeciesIn;
66  if (iSpeciesIn != iSpecies) {
67  UTIL_THROW("Error: iSpeciesIn != iSpecies");
68  }
69  in >> Label("nMolecule") >> nMolecule;
70  for (iMol = 0; iMol < nMolecule; ++iMol) {
71  in >> moleculeLabel >> iMolIn;
72  molPtr = &(simulation().getMolecule(iSpecies));
73  system().addMolecule(*molPtr);
74 
75  if (iMolIn != iMol) {
76  UTIL_THROW("Error: iMolIn != iMol");
77  }
78  if (molPtr != &system().molecule(iSpecies, iMol)) {
79  UTIL_THROW("Molecule index error");
80  }
81 
82  // If mutable, read molecule state id
83  if (speciesPtr->isMutable()) {
84  speciesPtr->mutator().readMoleculeState(in, *molPtr);
85  }
86 
87  // Read positions, shift into primary cell
88  for (molPtr->begin(atomIter); atomIter.notEnd(); ++atomIter) {
89  readAtom(in, *atomIter);
90  }
91 
92  }
93  }
94 
95  #ifdef SIMP_TETHER
96  { // Scope block for variables
97 
98  // Read Tethers
99  Vector anchor;
100  Atom* atomPtr;
101  int iTether, nTether, iAtom;
102  in >> Label("TETHERS");
103  in >> Label("nTether") >> nTether;
104  for (iTether = 0; iTether < nTether; ++iTether) {
105  in >> iSpecies >> iMol >> iAtom >> anchor;
106  if (iSpecies >= simulation().nSpecies()) {
107  UTIL_THROW("Invalid species index for tether");
108  }
109  if (iMol >= system().nMolecule(iSpecies)) {
110  UTIL_THROW("Invalid molecule index for tether");
111  }
112  molPtr = &system().molecule(iSpecies, iMol);
113  if (iAtom >= molPtr->nAtom()) {
114  Log::file() << iAtom << " " << molPtr->nAtom() << std::endl;
115  UTIL_THROW("Invalid atom index for tether");
116  }
117  atomPtr = &molPtr->atom(iAtom);
118  system().tetherMaster().addTether(*atomPtr, anchor);
119  }
120 
121  }
122  #endif
123 
124  #ifdef MCMD_LINK
125  { // Scope block for variables
126 
127  // Read Links
128  Atom* atom0Ptr;
129  Atom* atom1Ptr;
130  int iLink, nLink, iAtom, linkType;
131 
132  in >> Label("LINKS");
133  in >> Label("nLink") >> nLink;
134  for (iLink = 0; iLink < nLink; ++iLink) {
135 
136  // Read atom 0
137  in >> iSpecies >> iMol >> iAtom;
138  if (iSpecies < 0 || iSpecies >= simulation().nSpecies()) {
139  Log::file() << "iLink = " << iLink << std::endl;
140  UTIL_THROW("Invalid iSpecies0 index in link");
141  }
142  if (iMol < 0 || iMol >= system().nMolecule(iSpecies)) {
143  Log::file() << "iLink = " << iLink << std::endl;
144  UTIL_THROW("Invalid iMol0 index in link");
145  }
146  molPtr = &system().molecule(iSpecies, iMol);
147  if (iAtom < 0 || iAtom >= molPtr->nAtom()) {
148  Log::file() << "iLink = " << iLink << std::endl;
149  UTIL_THROW("Invalid iAtom0 index for link");
150  }
151  atom0Ptr = &molPtr->atom(iAtom);
152 
153  // Read atom 1
154  in >> iSpecies >> iMol >> iAtom;
155  if (iSpecies < 0 || iSpecies >= simulation().nSpecies()) {
156  Log::file() << "iLink = " << iLink << std::endl;
157  UTIL_THROW("Invalid iSpecies1 index in link");
158  }
159  if (iMol < 0 || iMol >= system().nMolecule(iSpecies)) {
160  Log::file() << "iLink = " << iLink << std::endl;
161  UTIL_THROW("Invalid iMol1 index in link");
162  }
163  molPtr = &system().molecule(iSpecies, iMol);
164  if (iAtom < 0 || iAtom >= molPtr->nAtom()) {
165  Log::file() << "iLink = " << iLink << std::endl;
166  UTIL_THROW("Invalid iAtom1 index for link");
167  }
168  atom1Ptr = &molPtr->atom(iAtom);
169 
170  in >> linkType;
171  system().linkMaster().addLink(*atom0Ptr, *atom1Ptr, linkType);
172 
173  }
174 
175  }

183  void McMdConfigIo::write(std::ostream &out)
184  {
185  using std::endl;
186 
187  // Write Boundary dimensions
188  out << "BOUNDARY" << endl << endl;
189  out << boundary() << endl;
190 
191  // Write atomic positions
192  System::ConstMoleculeIterator molIter;
193  Molecule::ConstAtomIterator atomIter;
194  Species* speciesPtr;
195  int iSpecies, iMolecule;
196  out << endl << "MOLECULES" << endl;
197  for (iSpecies = 0; iSpecies < simulation().nSpecies(); ++iSpecies) {
198  out << endl;
199  out << "species " << iSpecies << endl;
200  out << "nMolecule " << system().nMolecule(iSpecies) << endl;
201  speciesPtr = &simulation().species(iSpecies);
202  iMolecule = 0;
203  system().begin(iSpecies, molIter);
204  for ( ; molIter.notEnd(); ++molIter) {
205  out << endl;
206  out << "molecule " << iMolecule << endl;
207  if (speciesPtr->isMutable()) {
208  speciesPtr->mutator().writeMoleculeState(out, *molIter);
209  }
210  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
211  //out << atomIter->position() << endl;
212  writeAtom(out, *atomIter);
213  }
214  ++iMolecule;
215  }
216  }
217 
218  #ifdef SIMP_TETHER
219  { // Scope for local variables
220 
221  // Write Tethers
222  Tether* tetherPtr;
223  Atom* atomPtr;
224  Molecule* molPtr;
225  int iTether, nTether, iAtom;
226  out << std::endl;
227  out << "TETHERS" << endl << endl;
228  nTether = system().tetherMaster().nTether();
229  out << Label("nTether") << nTether << std::endl;
230  for (iTether = 0; iTether < nTether; ++iTether) {
231  tetherPtr = &(system().tetherMaster().tether(iTether));
232  atomPtr = &tetherPtr->atom();
233  molPtr = &atomPtr->molecule();
