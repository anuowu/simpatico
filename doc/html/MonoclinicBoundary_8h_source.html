530  double MonoclinicBoundary::distanceSq(const Vector &r1, const Vector &r2,
531  IntVector& translate) const
532  {
533 
534  double dx = r1[0] - r2[0];
535  if ( fabs(dx) > halfL_[0] ) {
536  if (dx > 0.0) {
537  dx -= l_[0];
538  translate[0] = -1;
539  } else {
540  dx += l_[0];
541  translate[0] = +1;
542  }
543  assert(fabs(dx) <= halfL_[0]);
544  } else {
545  translate[0] = 0;
546  }
547 
548  double dy = r1[1] - r2[1];
549  double dz = r1[2] - r2[2];
550  double u2 = dz + c3_*dy;
551  if ( fabs(dy) > halfL_[1] ) {
552  if (dy > 0.0) {
553  dy -= l_[1];
554  dz -= d_;
555  translate[1] = -1;
556  } else {
557  dy += l_[1];
558  dz += d_;
559  translate[1] = +1;
560  }
561  assert(fabs(dy) <= halfL_[1]);
562  } else {
563  translate[1] = 0;
564  }
565 
566  if ( fabs(u2) > halfL_[2]) {
567  if (u2 > 0.0) {
568  dz -= l_[2];
569  translate[2] = -1;
570  } else {
571  dz += l_[2];
572  translate[2] = +1;
573  }
574  assert(fabs(dz + c3_*dy) <= halfL_[2]);
575  } else {
576  translate[2] = 0;
577  }
578  return dx*dx+dy*dy+dz*dz;
