60  void
61  CfbEndBase::deleteEndAtom(Atom* endPtr, Atom* pvtPtr, int bondType,
62  double &rosenbluth, double &energy)
63  {
64  Vector bondVec;
65  Vector pvtPos = pvtPtr->position();
66  double trialEnergy, lengthSq, length;
67 
68  // Calculate bond length of pvt-end bond
69  lengthSq = boundary().distanceSq(pvtPos, endPtr->position());
70  length = sqrt(lengthSq);
71 
72  // Calculate current nonbonded pair energy of end atom
73  #ifndef SIMP_NOPAIR
74  energy = system().pairPotential().atomEnergy(*endPtr);
75  #else
76  energy = 0.0;
77  #endif
78 
79  #ifdef SIMP_ANGLE
80  AtomAngleArray angles;
81  const Angle *anglePtr;
82  const Atom *pvtPtr2(NULL);
83  Vector dr1, dr2;
84  int iAngle, angleTypeId(0);
85  double rsq1, rsq2, cosTheta;
86 
87  if (system().hasAnglePotential()) {
88 
89  // Get the angle type and pointers of atoms forming the angle.
90  getAtomAngles(*endPtr, angles);
91  for (iAngle = 0; iAngle < angles.size(); ++iAngle) {
92  anglePtr = angles[iAngle];
93  if (&anglePtr->atom(1) == pvtPtr) {
94  if (&anglePtr->atom(0) == endPtr) {
95  pvtPtr2 = &anglePtr->atom(2);
96  } else {
97  pvtPtr2 = &anglePtr->atom(0);
98  }
99  angleTypeId = anglePtr->typeId();
100  }
101  }
102 
103  // Calculate angle energy.
104  rsq1 = boundary().distanceSq(pvtPtr->position(),
105  pvtPtr2->position(), dr1);
106  rsq2 = boundary().distanceSq(endPtr->position(),
107  pvtPtr->position(), dr2);
108  cosTheta = dr1.dot(dr2) / sqrt(rsq1 * rsq2);
109 
110  energy += system().anglePotential().energy(cosTheta, angleTypeId);
111 
112  }
113  #endif
114 
115  #ifdef SIMP_EXTERNAL
116  if (system().hasExternalPotential()) {
117  energy += system().externalPotential().atomEnergy(*endPtr);
118  }
119  #endif
120 
121  // Rosenbluth factor = exp(-beta*(pair + angle + external))
122  rosenbluth = boltzmann(energy);
123 
124  // Add bond energy of current pvt-end bond to energy.
125  // This is the final value of inout energy parameter.
126  energy += system().bondPotential().energy(lengthSq, bondType);
127 
128  // Loop over nTrial - 1 additional trial positions:
129  for (int iTrial=0; iTrial < nTrial_ - 1; ++iTrial) {
130 
131  random().unitVector(bondVec);
132  bondVec *= length;
133  endPtr->position().add(pvtPos, bondVec);
134  boundary().shift(endPtr->position());
135 
136  #ifndef SIMP_NOPAIR
137  trialEnergy = system().pairPotential().atomEnergy(*endPtr);
138  #else
139  trialEnergy = 0.0;
140  #endif
141 
142  #ifdef SIMP_ANGLE
143  if (system().hasAnglePotential()) {
144 
145  // Get the angle type and atom pointer at the angle.
146  getAtomAngles(*endPtr, angles);
147  for (iAngle = 0; iAngle < angles.size(); ++iAngle) {
148  anglePtr = angles[iAngle];
149  if (&anglePtr->atom(1) == pvtPtr) {
150  if (&anglePtr->atom(0) == endPtr) {
151  pvtPtr2 = &anglePtr->atom(2);
152  } else {
153  pvtPtr2 = &anglePtr->atom(0);
154  }
155  angleTypeId = anglePtr->typeId();
156  }
157  }
158 
159  // Calculate energy.
160  rsq1 = boundary().distanceSq(pvtPtr->position(),
161  pvtPtr2->position(), dr1);
162  rsq2 = boundary().distanceSq(endPtr->position(),
163  pvtPtr->position(), dr2);
164  cosTheta = dr1.dot(dr2) / sqrt(rsq1 * rsq2);
165 
166  trialEnergy += system().anglePotential()
167  .energy(cosTheta, angleTypeId);
168  }
169  #endif
170 
171  #ifdef SIMP_EXTERNAL
172  if (system().hasExternalPotential()) {
173  trialEnergy +=
174  system().externalPotential().atomEnergy(*endPtr);
175  }
176  #endif
177 
178  rosenbluth += boltzmann(trialEnergy);
179  }

185  * Configuration bias algorithm for adding one atom to a chain end.
186  */
187  void
188  CfbEndBase::addEndAtom(Atom* endPtr, Atom* pvtPtr, int bondType,
189  double &rosenbluth, double &energy)
190  {
191  Vector trialPos[MaxTrial_];
192  Vector bondVec;
193  Vector pvtPos = pvtPtr->position();
194  double trialProb[MaxTrial_], trialEnergy[MaxTrial_];
195  double beta, length;
196  int iTrial;
197 
198  #ifdef SIMP_ANGLE
199  AtomAngleArray angles;
200  const Angle *anglePtr;
201  const Atom *pvtPtr2(NULL);
202  Vector dr1, dr2;
203  int iAngle, angleTypeId(0);
204  double rsq1, rsq2, cosTheta;
205  #endif
206 
207  // Generate a random bond length
208  beta = energyEnsemble().beta();
209  length =
210  system().bondPotential().randomBondLength(&random(), beta, bondType);
211 
212  // Loop over nTrial trial positions:
213  rosenbluth = 0.0;
214  for (iTrial=0; iTrial < nTrial_; ++iTrial) {
215  random().unitVector(bondVec);
216  bondVec *= length;
217  // trialPos = pvtPos + bondVec
218  trialPos[iTrial].add(pvtPos, bondVec);
219  boundary().shift(trialPos[iTrial]);
220  endPtr->position() = trialPos[iTrial];
221  #ifndef SIMP_NOPAIR
222  trialEnergy[iTrial] = system().pairPotential().atomEnergy(*endPtr);
223  #else
224  trialEnergy[iTrial] = 0.0;
225  #endif
226 
227  #ifdef SIMP_ANGLE
228  if (system().hasAnglePotential()) {
229 
230  getAtomAngles(*endPtr, angles);
231  for (iAngle = 0; iAngle < angles.size(); ++iAngle) {
232  anglePtr = angles[iAngle];
233  if (&anglePtr->atom(1) == pvtPtr) {
234  if (&anglePtr->atom(0) == endPtr) {
235  pvtPtr2 = &anglePtr->atom(2);
236  } else {
237  pvtPtr2 = &anglePtr->atom(0);
238  }
239  angleTypeId = anglePtr->typeId();
240  }
241  }
242 
243  // Get the angle spanned.
244  rsq1 = boundary().distanceSq(pvtPtr->position(),
245  pvtPtr2->position(), dr1);
246  rsq2 = boundary().distanceSq(endPtr->position(),
247  pvtPtr->position(), dr2);
248  cosTheta = dr1.dot(dr2) / sqrt(rsq1 * rsq2);
249 
250  trialEnergy[iTrial] += system().anglePotential().
251  energy(cosTheta, angleTypeId);
252  }
253  #endif
254 
255  #ifdef SIMP_EXTERNAL
256  if (system().hasExternalPotential()) {
257  trialEnergy[iTrial] +=
258  system().externalPotential().atomEnergy(*endPtr);
259  }
260  #endif
261 
262  trialProb[iTrial] = boltzmann(trialEnergy[iTrial]);
263  rosenbluth += trialProb[iTrial];
264  }
265 
266  // Normalize trial probabilities
267  for (iTrial = 0; iTrial < nTrial_; ++iTrial) {
268  trialProb[iTrial] = trialProb[iTrial]/rosenbluth;
269  }
270 
271  // Choose trial position
272  iTrial = random().drawFrom(trialProb, nTrial_);
273 
274  // Calculate total energy for chosen trial.
275  energy = system().bondPotential().energy(length*length, bondType);
276  energy += trialEnergy[iTrial];
277 
278  // Set position of new end atom to chosen value
279  endPtr->position() = trialPos[iTrial];
280 
281  }
