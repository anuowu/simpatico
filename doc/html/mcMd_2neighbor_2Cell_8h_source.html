110  /*
111  * Implementation notes:
112  *
113  * Pointers to atoms in a cell are stored in the atoms_[] array member.
114  * "Empty" elements of atoms_[] must contain a null (i.e., 0) pointer.
115  *
116  * The total number of atoms in the cell in is nAtomCell_. The array
117  * index of the first empty element in the atoms_[] array is given by
118  * firstEmptyPos_. All elements of atoms_[] with indices greater than
119  * or equal to firstClearPos_ are empty. An element atoms_[i] with
120  * i < firstEmptyPos_ may not be empty, one with i >= firstClearPos_ must
121  * be empty, and one with firstEmptyPos_ < i and i < firstClearPos_ may or
122  * may not be empty. When firstClearPos_ > 0, atoms_[firstClearPos_-1] may
123  * not be empty, since this would imply that firstClearPos_ is not the
124  * first in the contiguous block of empty elements. These conditions are
125  * checked in the isValid() method.
126  */

153  inline void Cell::deleteAtom(CellTag &cellTag)
154  {
155  int cellPos = cellTag.cellPos;
156 
157  // Delete atom from Cell member variables.
158  // Set atoms_[cellPos] to 0 to mark empty.
159  atoms_[cellPos] = 0;
160  nAtomCell_--;
161 
162  // Reset firstEmptyPos_
163  if (firstEmptyPos_ > cellPos) {
164  firstEmptyPos_ = cellPos;
165  }
166 
167  // Reset firstClearPos_
168  int i = firstClearPos_ - 1;
169  while (i >= 0 && atoms_[i] == 0) {
170  i--;
171  }
172  firstClearPos_ = i + 1;
173 
174  // Reset cellTag cellId and cellPos to negative integer values
175  // to indicate that the atom is no longer associated with a Cell.
176  cellTag.cellId = NullIndex;
177  cellTag.cellPos = NullIndex;

184  inline void Cell::addAtom(CellTag &cellTag, Atom &atom, int cellId)
185  {
186 
187  // If cell is already full, throw Exception
188  if (firstEmptyPos_ == MaxAtomCell) {
189  UTIL_THROW("Too many atoms in one cell");
190  }
191 
192  // Add atom to Cell
193  atoms_[firstEmptyPos_] = &atom;
194  ++nAtomCell_;
195 
196  // Record cell in cellTag
197  cellTag.cellId = cellId;
198  cellTag.cellPos = firstEmptyPos_;
199 
200  // Reset firstEmptyPos_ and firstClearPos_
201  if (firstEmptyPos_ < MaxAtomCell - 1) {
202 
203  if (firstEmptyPos_ == firstClearPos_) {
204 
205  firstEmptyPos_ = firstEmptyPos_ + 1;
206  firstClearPos_ = firstClearPos_ + 1;
207 
208  } else {
209 
210  // Find the next empty element, for which atoms_[i] is null
211  for (int i = firstEmptyPos_ + 1; i < MaxAtomCell; ++i) {
212  if (atoms_[i] == 0) {
213  firstEmptyPos_ = i;
214  break;
215  }
216  }
217 
218  }
219 
220  } else {
221 
222  // If firstEmptyPos_ was MaxAtomCell - 1, no empty slots remain.
223  // Set firstEmptyPos_ = MaxAtomCell to indicate this condition.
224  // Attempting to add one more atom will trigger an Exception.
225  firstEmptyPos_ = MaxAtomCell; // Note: Invalid array index
226  firstClearPos_ = MaxAtomCell; // Note: Invalid array index
227 
228  }
