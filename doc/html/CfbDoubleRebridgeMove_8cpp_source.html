105  /*
106  * Generate, attempt and accept or reject a Monte Carlo move.
107  *
108  * A complete move involve the following stepts:
109  *
110  * 1) choose a molecule-i randomly;
111  * 2) find molecule-j of the same type as i, and monomer pairs satisfying
112  * the distance criterion exist;
113  * 3) Delete the two rebridging monomers; first molecule-i, then -j.
114  * 4) Rebuild molecule-j, then -i. (in the JCP paper, the order is chosen
115  * at random.)
116  *
117  */
118  bool CfbDoubleRebridgeMove::move()
119  {
120  bool found, accept;
121  int iMol, jMol, sign, beginId, nEnd, i;
122  Molecule *iPtr; // pointer to i-th molecule
123  Molecule *jPtr; // pointer to j-th molecule
124  int *bonds; // bond types of a consequtive blocks of atoms
125  Atom *thisPtr; // pointer to the current end atom
126  Atom *tempPtr; // pointer used for swap atom positions
127  double rosenbluth, rosen_r, rosen_f;
128  double energy, energy_r, energy_f;
129  double po2n, pn2o;
130  Vector swapPos;
131 
132  incrementNAttempt();
133 
134  // Choose a random direction
135  if (random().uniform(0.0, 1.0) > 0.5)
136  sign = +1;
137  else
138  sign = -1;
139 
140  // Search for a pair of candidate sites and calculate probability
141  found = forwardScan(sign, iMol, jMol, beginId, po2n);
142  if (!found) return found;
143 
144  iPtr = &(system().molecule(speciesId_, iMol));
145  jPtr = &(system().molecule(speciesId_, jMol));
146 
147  // Save positions for atoms to be regrown
148  thisPtr = &(iPtr->atom(beginId));
149  tempPtr = &(jPtr->atom(beginId));
150  for (i = 0; i < nRegrow_; ++i) {
151  iOldPos_[i] = thisPtr->position();
152  jOldPos_[i] = tempPtr->position();
153 
154  thisPtr += sign;
155  tempPtr += sign;
156  }
157 
158  // Prepare for the rebridge move
159  accept = false;
160 
161  bonds = new int[nRegrow_ + 1];
162  if (sign == +1) {
163  for (i = 0; i < nRegrow_ + 1; ++i)
164  bonds[i] = iPtr->bond(beginId - 1 + nRegrow_ - i).typeId();
165  } else {
166  for (i = 0; i < nRegrow_ + 1; ++i)
167  bonds[i] = iPtr->bond(beginId - nRegrow_ + i).typeId();
168  }
169 
170  // Deleting atoms: first i-th molecule, then j-th
171  rosen_r = 1.0;
172  energy_r = 0.0;
173 
174  thisPtr = &(iPtr->atom(beginId + (nRegrow_-1)*sign));
175  deleteSequence(nRegrow_, sign, thisPtr, bonds, rosenbluth, energy);
176  rosen_r *= rosenbluth;
177  energy_r += energy;
178 
179  thisPtr = &(jPtr->atom(beginId + (nRegrow_-1)*sign));
180  deleteSequence(nRegrow_, sign, thisPtr, bonds, rosenbluth, energy);
181  rosen_r *= rosenbluth;
182  energy_r += energy;
183 
184  // Swap the atom positions of the dangling end
185  if (sign == +1)
186  nEnd = iPtr->nAtom() - beginId - nRegrow_;
187  else
188  nEnd = beginId + 1 - nRegrow_;
189 
190  thisPtr = &(iPtr->atom(beginId + nRegrow_*sign));
191  tempPtr = &(jPtr->atom(beginId + nRegrow_*sign));
192  for (i = 0; i < nEnd; ++i) {
193  swapPos = thisPtr->position();
194  //system().moveAtom(*thisPtr, tempPtr->position());
195  thisPtr->position() = tempPtr->position();
196  #ifndef INTER_NOPAIR
197  system().pairPotential().updateAtomCell(*thisPtr);
198  #endif
199  //system().moveAtom(*tempPtr, swapPos);
200  tempPtr->position() = swapPos;
201  #ifndef INTER_NOPAIR
202  system().pairPotential().updateAtomCell(*tempPtr);
203  #endif
204 
205  thisPtr += sign;
206  tempPtr += sign;
207  }
208 
209  // Regrow atoms: first j-th molecule, then i-th
210  rosen_f = 1.0;
211  energy_f = 0.0;
212 
213  thisPtr = &(jPtr->atom(beginId));
214  addSequence(nRegrow_, sign, thisPtr, bonds, rosenbluth, energy);
215  rosen_f *= rosenbluth;
216  energy_f += energy;
217 
218  thisPtr = &(iPtr->atom(beginId));
219  addSequence(nRegrow_, sign, thisPtr, bonds, rosenbluth, energy);
220  rosen_f *= rosenbluth;
221  energy_f += energy;
222 
223  // Reverse scan
224  found = reverseScan(sign, iMol, jMol, beginId - sign, pn2o);
225 
226  // Decide whether to accept or reject
227  if (found)
228  accept = random().metropolis(rosen_f * pn2o / rosen_r / po2n);
229  else
230  accept = false;
231 
232  if (accept) {
233 
234  // Increment counter for accepted moves of this class.
235  incrementNAccept();
236 
237  } else {
238 
239  // If the move is rejected, restore nRegrow_ positions
240  thisPtr = &(iPtr->atom(beginId));
241  tempPtr = &(jPtr->atom(beginId));
242  for (i = 0; i < nRegrow_; ++i) {
243  //system().moveAtom(*thisPtr, iOldPos_[i]);
244  thisPtr->position() = iOldPos_[i];
245  #ifndef INTER_NOPAIR
246  system().pairPotential().updateAtomCell(*thisPtr);
247  #endif
248  //system().moveAtom(*tempPtr, jOldPos_[i]);
249  tempPtr->position() = jOldPos_[i];
250  #ifndef INTER_NOPAIR
251  system().pairPotential().updateAtomCell(*tempPtr);
252  #endif
253 
254  thisPtr += sign;
255  tempPtr += sign;
256  }
257 
258  // Restore atom positions at the dangling end
259  thisPtr = &(iPtr->atom(beginId + nRegrow_*sign));
260  tempPtr = &(jPtr->atom(beginId + nRegrow_*sign));
261  for (i = 0; i < nEnd; ++i) {
262  swapPos = thisPtr->position();
263  //system().moveAtom(*thisPtr, tempPtr->position());
264  thisPtr->position() = tempPtr->position();
265  #ifndef INTER_NOPAIR
266  system().pairPotential().updateAtomCell(*thisPtr);
267  #endif
268  //system().moveAtom(*tempPtr, swapPos);
269  tempPtr->position() = swapPos;
270  #ifndef INTER_NOPAIR
271  system().pairPotential().updateAtomCell(*tempPtr);
272  #endif
273 
274  thisPtr += sign;
275  tempPtr += sign;
276  }
277 
278  }
279 
280  // Release memory
281  delete [] bonds;
282 
283  return accept;
284  }

286  /*
287  * Scanning chains to find all possible pairs of rebriding sites satisfying
288  * the distance criterion, then choose one randomly to commit the double
289  * rebridging move.
290  *
291  * probability = 1 / (number of pairs satisfying the distance criterion)
292  *
293  * rebriding criterion:
294  *
295  * distance(i1,j2) < bridgeLength
296  * && distance(i2,j1) < bridgeLength
297  *
298  * i1 i2
299  * -->------o-o-o---- i-chain
300  * \ /
301  * X
302  * / \
303  * -->------o-o-o---- j-chain
304  * j1 j2
305  *
306  *
307  */
308  bool CfbDoubleRebridgeMove::forwardScan(int sign, int &iMol,
309  int &jMol, int &beginId, double &prob)
310  {
311  Molecule *iPtr, *jPtr;
312  Vector iPos1, jPos1, iPos2, jPos2;
313  int nMol, nAtom, maxPair;
314  int signRegrow, nPairs, k, choice;
315  int *molBuf, *atomBuf;
316  double lengthSq, bridgeLengthSq;
317  bool found;
318 
319  // Number of molecules
320  nMol = system().nMolecule(speciesId_);
321 
322  // Randomly choose one molecule
323  iPtr = &(system().randomMolecule(speciesId_));
324  iMol = system().moleculeId(*iPtr);
325 
326  // Number of atoms per molecule
327  nAtom = iPtr->nAtom();
328 
329  // Require that chain length - 2 >= nRegrow_
330  if (nRegrow_ > nAtom - 2) {
331  UTIL_THROW("nRegrow_ >= chain length");
332  }
333 
334  // Allocate arrays holding candidate atom pairs
335  maxPair = (nMol - 1) * (nAtom - 1 - nRegrow_);
336  molBuf = new int[maxPair];
337  atomBuf = new int[maxPair];
338 
339  // Identify the direction of scaning
340  beginId = (sign == -1) ? (nAtom-1) : 0;
341 
342  // Auxiliary variables
343  signRegrow = sign * (nRegrow_ + 1);
344  bridgeLengthSq = bridgeLength_ * bridgeLength_;
345 
346  // Nullify counters
347  nPairs = 0;
348  found = false;
349 
350  // Loop over j molecules to scan possible pairs
351  for (jMol = 0; jMol < nMol; ++jMol) {
352 
353  jPtr = &(system().molecule(speciesId_, jMol));
354 
355  if (jMol != iMol) {
356 
357  for (k = 0; k < nAtom - 1 - nRegrow_; ++k) {
358 
359  // Check if i1 to j2 bridging is possible
360  iPos1 = iPtr->atom(beginId + k*sign).position();
361  jPos2 = jPtr->atom(beginId + signRegrow + k*sign).position();
362  lengthSq = boundary().distanceSq(iPos1, jPos2);
363 
364  if (lengthSq < bridgeLengthSq) {
365  // Check if j2 to i1 bridging is possible
366  jPos1 = jPtr->atom(beginId + k*sign).position();
367  iPos2 = iPtr->atom(beginId + signRegrow + k*sign).position();
368  lengthSq = boundary().distanceSq(iPos2, jPos1);
369 
370  if (lengthSq < bridgeLengthSq) {
371  molBuf[nPairs] = jMol;
372  atomBuf[nPairs] = beginId + k*sign;
373  nPairs += 1;
374  }
375  }
376 
377  } // loop over atom positions
378  } // if jMol != iMol
379  } // loop over j molecules
380 
381  if (nPairs > 0) found = true;
382 
383  if (found) {
384  // iMol is not changed after the random selection
385  choice = random().uniformInt(0, nPairs);
386  jMol = molBuf[choice];
387  beginId = atomBuf[choice] + sign; // return the the active atom ID
388  prob = 1.0 / double(nPairs);
389  } else {
390  iMol = -1; // invalid value
391  jMol = -1; // invalid value
392  beginId = -1; // invalid value
393  prob = 0.0;
394  }
395 
396  // Deallocate memory and quit
397  delete [] molBuf;
398  delete [] atomBuf;
399  return found;
400 
401  }

403  /*
404  * Scanning chains to find the probability of choosing the specific reverse
405  * move.
406  * probability = 1 / (number of pairs satisfying the distance criterion)
407  *
408  * This method is different than the forward one in two aspects:
409  * (1) It first check if the old configuration is one possible reverse
410  * move; and return "0" probability if not.
411  * (2) It does not store the indices of possible monomer pairs and simply
412  * count the number of all possible bridging sites.
413  */
414  bool CfbDoubleRebridgeMove::reverseScan(int sign, int iMol,
415  int jMol, int beginId, double &prob)
416  {
417  Molecule *iPtr, *jPtr;
418  Vector iPos1, jPos1, iPos2, jPos2;
419  int nMol, nAtom, j;
420  int signRegrow, nPairs, headId, k;
421  double lengthSq, bridgeLengthSq;
422  bool found;
423 
424  // Retrieve molecule pointers
425  iPtr = &(system().molecule(speciesId_, iMol));
426  jPtr = &(system().molecule(speciesId_, jMol));
427 
428  // Number of molecules
429  nMol = system().nMolecule(speciesId_);
430 
431  // Number of atoms per molecule
432  nAtom = iPtr->nAtom();
433 
434  // Auxiliary variables
435  signRegrow = sign * (nRegrow_ + 1);
436  bridgeLengthSq = bridgeLength_ * bridgeLength_;
437 
438  // Initialize returning variable
439  found = false;
440 
441  // Check if i1 to j2 bridging is possible
442  iPos1 = iPtr->atom(beginId).position();
443  jPos2 = jPtr->atom(beginId + signRegrow).position();
444  lengthSq = boundary().distanceSq(iPos1, jPos2);
445 
446  if (lengthSq < bridgeLengthSq) {
447  // Check if j2 to i1 bridging is possible
448  jPos1 = jPtr->atom(beginId).position();
449  iPos2 = iPtr->atom(beginId + signRegrow).position();
450  lengthSq = boundary().distanceSq(iPos2, jPos1);
451 
452  if (lengthSq < bridgeLengthSq) {
453  found = true;
454  } else {
455  prob = 0.0;
456  return found;
457  }
458  } else {
459  prob = 0.0;
460  return found;
461  }
462 
463  // Auxiliary variables
464  headId = (sign == -1) ? (nAtom-1) : 0;
465  iMol = system().moleculeId(*iPtr);
466 
467  // Nullify counters
468  nPairs = 0;
469 
470  // Loop over j molecules to scan possible pairs
471  for (j = 0; j < nMol; ++j) {
472 
473  jPtr = &(system().molecule(speciesId_, j));
474 
475  if (j != iMol) {
476 
477  for (k = 0; k < nAtom - 1 - nRegrow_; ++k) {
478 
479  // Check if i1 to j2 bridging is possible
480  iPos1 = iPtr->atom(headId + k*sign).position();
481  jPos2 = jPtr->atom(headId + signRegrow + k*sign).position();
482  lengthSq = boundary().distanceSq(iPos1, jPos2);
483 
484  if (lengthSq < bridgeLengthSq) {
485  // Check if j2 to j1 bridging is possible
486  jPos1 = jPtr->atom(headId + k*sign).position();
487  iPos2 = iPtr->atom(headId + signRegrow + k*sign).position();
488  lengthSq = boundary().distanceSq(iPos2, jPos1);
489 
490  if (lengthSq < bridgeLengthSq) nPairs += 1;
491  }
492 
493  } // loop over atom positions
494  } // if j != iMol
495  } // loop over j molecules
496 
497  // nPairs is always greater than or equal to 1
498  prob = 1.0 / double(nPairs);
499  return found;
500  }
