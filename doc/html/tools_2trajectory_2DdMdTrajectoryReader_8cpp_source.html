44  bool DdMdTrajectoryReader::readFrame(std::ifstream& file)
45  {
46  BinaryFileIArchive ar(file);
47 
48  // Attempt to read iStep
49  long iStep = -1;
50  ar >> iStep;
51 
52  // Return false if read failed, indicating end of file.
53  if (file.eof()) {
54  return false;
55  }
56 
57  // Read boundary dimensions
58  Util::Boundary& boundary = configuration().boundary();
59  ar >> boundary;
60 
61  // Loop over atoms, read atomic positions
62  AtomStorage* storagePtr = &configuration().atoms();
63  Atom* atomPtr;
64  Vector r;
65  double h = 1.0/(double(UINT_MAX) + 1.0);
66  int id, i, j;
67  unsigned int ir;
68  for (i = 0; i < nAtom_; ++i) {
69  ar >> id;
70  atomPtr = storagePtr->ptr(id);
71  if (atomPtr == 0) {
72  UTIL_THROW("Unknown atom");
73  }
74  for (j = 0; j < Dimension; ++j) {
75  ar >> ir;
76  r[j] = ir*h;
77  }
78  boundary.transformGenToCart(r, atomPtr->position);
79  }
80 
81  return true;
82  }
