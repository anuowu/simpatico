117  void McNVTChemicalPotential::sample(long iStep)
118  {
119  if (isAtInterval(iStep)) {
120 
121  Molecule* molPtr;
122  Atom* endPtr, *pvt1Ptr;
123  Vector trialPos[MaxTrial_], bondVec, pvt1Pos;
124  #ifdef SIMP_ANGLE
125  double angle;
126  Atom* pvt2Ptr;
127  Vector pvt2Pos;
128  #endif
129  double beta, length;
130  // Rosenbluth factor and energy of inserted bonds at each stage.
131  double w = 0, de = 0;
132  // Total rosenbluth factor and energy of inserted polymer.
133  double rosenbluth = 1, energy = 0;
134  double trialProb[MaxTrial_], trialEnergy[MaxTrial_];
135  int iTrial;
136 
137  beta = energyEnsemble().beta();
138  molPtr = &(simulation().getMolecule(speciesId_));
139  system().addMolecule(*molPtr);
140 
141  // Main molecule inserting loop:
142  for (int i = 0; i < nMoleculeTrial_; i++) {
143 
144  // Inserting the first atom.
145  endPtr = &molPtr->atom(0);
146  for (iTrial = 0; iTrial < nTrial_; ++iTrial) {
147  // trialPos = pvtPos + bondVec
148  boundary().randomPosition(random(), trialPos[iTrial]);
149  boundary().shift(trialPos[iTrial]);
150 
151  #ifndef SIMP_NOPAIR
152  trialEnergy[iTrial] =
153  system().pairPotential().atomEnergy(*endPtr);
154  #else
155  trialEnergy[iTrial] = 0.0;
156  #endif
157 
158  #ifdef SIMP_EXTERNAL
159  trialEnergy[iTrial] +=
160  system().externalPotential().atomEnergy(*endPtr);
161  #endif
162 
163  trialProb[iTrial] = boltzmann(trialEnergy[iTrial]);
164  w += trialProb[iTrial];
165  }
166 
167  // Normalize trial probabilities
168  for (iTrial = 0; iTrial < nTrial_; ++iTrial) {
169  trialProb[iTrial] = trialProb[iTrial]/rosenbluth;
170  }
171 
172  // Choose trial position
173  iTrial = random().drawFrom(trialProb, nTrial_);
174 
175  // Calculate total energy for chosen trial
176  de = trialEnergy[iTrial];
177 
178  // Set position of new end atom to chosen trialPos Vector
179  endPtr->position() = trialPos[iTrial];
180 
181  #ifndef SIMP_NOPAIR
182  system().pairPotential().addAtom(*endPtr);
183  #endif
184 
185  rosenbluth *= w;
186  energy += de;
187 
188  // Inserting the second atom.
189  int bondTypeId = molPtr->bond(0).typeId();
190  length = system().bondPotential().randomBondLength(&random(),
191  beta,
192  bondTypeId);
193 
194  w = 0;
195  de = 0;
196 
197  pvt1Ptr = &molPtr->atom(0);
198  endPtr = &molPtr->atom(1);
199 
200  pvt1Pos = pvt1Ptr->position();
201 
202  for (iTrial = 0; iTrial < nTrial_; ++iTrial) {
203  // trialPos = pvtPos + bondVec
204  random().unitVector(bondVec);
205  bondVec *= length;
206  // trialPos = pvtPos + bondVec
207  trialPos[iTrial].add(pvt1Pos, bondVec);
208  boundary().shift(trialPos[iTrial]);
209  endPtr->position() = trialPos[iTrial];
210 
211  #ifndef SIMP_NOPAIR
212  trialEnergy[iTrial] =
213  system().pairPotential().atomEnergy(*endPtr);
214  #else
215  trialEnergy[iTrial] = 0.0;
216  #endif
217 
218  #ifdef SIMP_EXTERNAL
219  trialEnergy[iTrial] +=
220  system().externalPotential().atomEnergy(*endPtr);
221  #endif
222 
223  trialProb[iTrial] = boltzmann(trialEnergy[iTrial]);
224  w += trialProb[iTrial];
225  }
226 
227  // Normalize trial probabilities
228  for (iTrial = 0; iTrial < nTrial_; ++iTrial) {
229  trialProb[iTrial] = trialProb[iTrial]/rosenbluth;
230  }
231 
232  // Choose trial position
233  iTrial = random().drawFrom(trialProb, nTrial_);
234 
235  // Fetch nonbonded energy for chosen trial
236  de = trialEnergy[iTrial];
237 
238  // Add bond energy
239  de += system().bondPotential().energy(length*length, bondTypeId);
240 
241  // Set position of new end atom to chosen trialPos Vector
242  endPtr->position() = trialPos[iTrial];
243 
244  #ifndef SIMP_NOPAIR
245  system().pairPotential().addAtom(*endPtr);
246  #endif
247 
248  rosenbluth *= w;
249  energy += de;
250 
251  // Inserting the third atom.
252  bondTypeId = molPtr->bond(1).typeId();
253  length =
254  system().bondPotential().randomBondLength(&random(),
255  beta, bondTypeId);
256 
257  #ifdef SIMP_ANGLE
258  int angleTypeId = molPtr->angle(0).typeId();
259  angle =
260  system().anglePotential().randomAngle(&random(),
261  beta, angleTypeId);
262  pvt2Ptr = &molPtr->atom(0);
263  pvt2Pos = pvt2Ptr->position();
264  Vector n = pvt1Pos;
265  n -= pvt2Pos;
266  #endif
267 
268  w = 0;
269  de = 0;
270  pvt1Ptr = &molPtr->atom(1);
271  pvt1Pos = pvt1Ptr->position();
272  endPtr = &molPtr->atom(2);
273 
274  for (iTrial = 0; iTrial < nTrial_; ++iTrial) {
275  // trialPos = pvtPos + bondVec
276  #ifdef SIMP_ANGLE
277  if (system().hasAnglePotential()) {
278  uniformCone(length, angle, n, bondVec);
279  } else {
280  random().unitVector(bondVec);
281  bondVec *= length;
282  }
283  #else
284  random().unitVector(bondVec);
285  bondVec *= length;
286  #endif
287 
288  // trialPos = pvtPos + bondVec
289  trialPos[iTrial].add(pvt1Pos, bondVec);
290  boundary().shift(trialPos[iTrial]);
291  endPtr->position() = trialPos[iTrial];
292 
293  #ifndef SIMP_NOPAIR
294  trialEnergy[iTrial] =
295  system().pairPotential().atomEnergy(*endPtr);
296  #else
297  trialEnergy[iTrial] = 0.0;
298  #endif
299 
300  #ifdef SIMP_EXTERNAL
301  trialEnergy[iTrial] +=
302  system().externalPotential().atomEnergy(*endPtr);
303  #endif
304 
305  trialProb[iTrial] = boltzmann(trialEnergy[iTrial]);
306  w += trialProb[iTrial];
307  }
308  // Normalize trial probabilities
309  for (iTrial = 0; iTrial < nTrial_; ++iTrial) {
310  trialProb[iTrial] = trialProb[iTrial]/rosenbluth;
311  }
312 
313  // Choose trial position
314  iTrial = random().drawFrom(trialProb, nTrial_);
315 
316  // Get nonbonded energy for chosen trial
317  de = trialEnergy[iTrial];
318 
319  // Add bonded energies
320  de += system().bondPotential().energy(length*length, bondTypeId);
321  #ifdef SIMP_ANGLE
322  de += system().anglePotential().energy(cos(angle), angleTypeId);
323  #endif
324 
325  // Set position of new end atom to chosen trialPos Vector
326  endPtr->position() = trialPos[iTrial];
327 
328  #ifndef SIMP_NOPAIR
329  system().pairPotential().addAtom(*endPtr);
330  #endif
331 
332  rosenbluth *= w;
333  energy += de;
334 
335  for (int atomId = 3; atomId < molPtr->nAtom(); ++atomId) {
336  addEndAtom(molPtr, atomId, w, de);
337  rosenbluth *= w;
338  energy += de;
339  #ifndef SIMP_NOPAIR
340  system().pairPotential().addAtom(molPtr->atom(atomId));
341  #endif
342  }
343 
344  rosenbluth = rosenbluth / pow(nTrial_,molPtr->nAtom());
345  accumulator_.sample(rosenbluth, outputFile_);
346 
347  #ifndef SIMP_NOPAIR
348  for (int atomId = 0; atomId < molPtr->nAtom(); ++atomId) {
349  system().pairPotential().deleteAtom(molPtr->atom(atomId));
350  }
351  #endif
352  }
353 
354  system().removeMolecule(*molPtr);
355  simulation().returnMolecule(*molPtr);
356 
357  }

408  /*
409  * Configuration bias algorithm for adding one atom to a chain end.
410  */
411  void
412  McNVTChemicalPotential::addEndAtom(Molecule* molPtr,
413  int atomId, double &rosenbluth,
414  double &energy)
415  {
416  Atom* endPtr = &molPtr->atom(atomId);
417  Vector trialPos[MaxTrial_];
418  Vector bondVec;
419  Vector pvt1Pos = molPtr->atom(atomId-1).position();
420 
421  #ifdef SIMP_ANGLE
422  Vector pvt2Pos = molPtr->atom(atomId-2).position();
423  #endif
424 
425  #ifdef SIMP_DIHEDRAL
426  Vector pvt3Pos = molPtr->atom(atomId-3).position();
427  #endif
428 
429  double trialProb[MaxTrial_], trialEnergy[MaxTrial_];
430  double beta, length;
431  int bondTypeId = molPtr->bond(atomId - 1).typeId();
432 
433  #ifdef SIMP_ANGLE
434  double angle;
435  int angleTypeId = molPtr->angle(atomId - 2).typeId();
436  #endif
437 
438  int iTrial;
439  double eMin = 0;
440 
441  // Generate a random bond-length and bond-angle.
442  beta = energyEnsemble().beta();
443  length = system().bondPotential().randomBondLength(&random(), beta,
444  molPtr->bond(atomId-1).typeId());
445  #ifdef SIMP_ANGLE
446  Vector n = pvt1Pos;
447  n -= pvt2Pos;
448  angle = system().anglePotential().randomAngle(&random(), beta,
449  molPtr->angle(atomId-2).typeId());
450  #endif
451  #ifdef SIMP_DIHEDRAL
452  Vector dR1 = pvt3Pos;
453  Vector dR2 = pvt2Pos;
454  Vector dR3 = pvt1Pos;
455  dR1 -= pvt2Pos;
456  dR2 -= pvt1Pos;
457  dR3 -= endPtr->position();
458  #endif
459 
460  // Loop over nTrial trial positions:
461  rosenbluth = 0.0;
462  for (iTrial=0; iTrial < nTrial_; ++iTrial) {
463 
464  #ifdef SIMP_ANGLE
465  if (system().hasAnglePotential()) {
466  uniformCone(length, angle, n, bondVec);
467  } else {
468  random().unitVector(bondVec);
469  bondVec *= length;
470  }
471  #else
472  random().unitVector(bondVec);
473  bondVec *= length;
474  #endif
475 
476  trialPos[iTrial].add(pvt1Pos, bondVec);
477  boundary().shift(trialPos[iTrial]);
478  endPtr->position() = trialPos[iTrial];
479 
480  #ifndef SIMP_NOPAIR
481  trialEnergy[iTrial] = system().pairPotential().atomEnergy(*endPtr);
482  #else
483  trialEnergy[iTrial] = 0.0;
484  #endif
485 
486  #ifdef SIMP_DIHEDRAL
487  if (system().hasDihedralPotential()) {
488  trialEnergy[iTrial] +=
489  system().dihedralPotential().energy(dR1, dR2, dR3,
490  molPtr->dihedral(atomId-3).typeId());
491  }
492  #endif
493 
494  #ifdef SIMP_EXTERNAL
495  trialEnergy[iTrial] +=
496  system().externalPotential().atomEnergy(*endPtr);
497  #endif
498 
499  // Finding the Minimum of the trialEnergy vector
500  if (eMin > trialEnergy[iTrial])
501  eMin = trialEnergy[iTrial];
502  Eaccumulator_.sample(trialEnergy[iTrial]);
503 
504  trialProb[iTrial] = boltzmann(trialEnergy[iTrial]);
505  rosenbluth += trialProb[iTrial];
506  }
507 
508  Emaccumulator_.sample(eMin);
509  BRaccumulator_.sample(rosenbluth);
510  // Normalize trial probabilities
511  for (iTrial = 0; iTrial < nTrial_; ++iTrial) {
512  trialProb[iTrial] = trialProb[iTrial]/rosenbluth;
513  }
514 
515  // Choose trial position
516  iTrial = random().drawFrom(trialProb, nTrial_);
517 
518  // Calculate total energy for chosen trial.
519  energy = system().bondPotential().energy(length*length, bondTypeId);
520  #ifdef SIMP_ANGLE
521  energy += system().anglePotential().energy(cos(angle), angleTypeId);
522  #endif
523 
524  energy += trialEnergy[iTrial];
525  Ecaccumulator_.sample(energy);
526 
527  // Set position of new end atom to chosen value
528  endPtr->position() = trialPos[iTrial];
529  }

531  /*
532  * Finding a vector which make an angle \theta with vector n and has length b.
533  */
534  void McNVTChemicalPotential::uniformCone(const double length,
535  const double angle,
536  const Vector n, Vector &p)
537  {
538  Vector v1;
539 
540  do {
541  random().unitVector(p);
542  v1.divide(n,n.abs());
543  v1.multiply(v1,p.projection(v1));
544  p -= v1;
545  } while (p.abs() > 1.0E-8);
546 
547  v1.divide(n,n.abs());
548  v1 *= length*cos(angle);
549  p /= p.abs();
550  p *= length*sin(angle);
551  p += (v1);
552  }
